\section{Preliminares e Notaﺣ۶ﺣ۲o}
% ============================================================

Denotamos por $\Gnp$ o grafo aleatﺣﺏrio binomial com $n$ vﺣ۸rtices em que cada aresta
estﺣ۰ presente independentemente com probabilidade $p = p(n)$.
Dizemos que $\Gnp$ satisfaz uma propriedade \emph{assintoticamente quase certamente}
(\whp) se a probabilidade de satisfazﺣ۹-la tende a $1$ quando $n \to \infty$.

\begin{definition}[Limiar e tipos de limiar]
Seja $\mathcal{P}$ uma propriedade monﺣﺏtona crescente de grafos.
Uma funﺣ۶ﺣ۲o $\hat{p} = \hat{p}(n)$ ﺣ۸ um \textbf{limiar grosseiro} (\emph{coarse threshold})
para $\mathcal{P}$ em $\Gnp$ se
\[
  \lim_{n\to\infty} \Pr[\Gnp \in \mathcal{P}] =
  \begin{cases}
    0 & \text{se } p = o(\hat{p}),\\
    1 & \text{se } p = \omega(\hat{p}).
  \end{cases}
\]
Dizemos que $\hat{p}$ ﺣ۸ um \textbf{limiar semi-agudo} (\emph{semi-sharp threshold})
se existem constantes $c, C > 0$ tais que
\[
  \lim_{n\to\infty} \Pr[\Gnp \in \mathcal{P}] =
  \begin{cases}
    0 & \text{se } p \le c\,\hat{p},\\
    1 & \text{se } p \ge C\,\hat{p}.
  \end{cases}
\]
O \textbf{$0$-enunciado} refere-se ﺣ  afirmaﺣ۶ﺣ۲o para $p$ abaixo do limiar,
e o \textbf{$1$-enunciado} ﺣ  afirmaﺣ۶ﺣ۲o para $p$ acima do limiar.
\end{definition}

\begin{definition}[Densidades de um grafo]
\label{def:densidades}
Dado um grafo $H$, definem-se:
\begin{itemize}
  \item A \textbf{densidade} de $H$:
    $d(H) := \dfrac{e(H)}{v(H)}$, onde $e(H) = |E(H)|$ e $v(H) = |V(H)|$.

  \item A \textbf{densidade mﺣ۰xima} de $H$:
    \[
      \maxdens(H) := \max\!\left\{\frac{e(J)}{v(J)} : J \subseteq H,\; v(J) \ge 1\right\}.
    \]
    Dizemos que $H$ ﺣ۸ \textbf{balanceado} se $\maxdens(H) = d(H)$, e
    \textbf{estritamente balanceado} se $\maxdens(H) > d(J)$ para todo
    $J \subsetneq H$ com $v(J) \ge 1$.

  \item A \textbf{2-densidade} de $H$ (para $v(H) \ge 3$):
    \[
      d_2(H) := \frac{e(H)-1}{v(H)-2}.
    \]

  \item A \textbf{2-densidade mﺣ۰xima} de $H$:
    \[
      \mdens(H) := \max\!\left\{d_2(J) : J \subseteq H,\; v(J) \ge 3\right\}
        \cup \left\{\tfrac{1}{2}\right\}.
    \]
    (A fraﺣ۶ﺣ۲o $\frac{1}{2}$ corresponde ao caso $H \cong K_2$.)
    Dizemos que $H$ ﺣ۸ \textbf{$2$-balanceado} se $\mdens(H) = d_2(H)$, e
    \textbf{estritamente $2$-balanceado} se $\mdens(H) > d_2(J)$ para todo
    $J \subsetneq H$.
\end{itemize}
\end{definition}

\begin{example}
  Para o clique $K_k$ com $k \ge 3$:
  $\mdens(K_k) = \dfrac{\binom{k}{2}-1}{k-2} = \dfrac{k+1}{2}$.
  Para o ciclo $C_\ell$ com $\ell \ge 3$:
  $\mdens(C_\ell) = \dfrac{\ell-1}{\ell-2}$.
  Em particular,
  $\mdens(K_5) = 3$, $\mdens(C_5) = \frac{4}{3}$, $\mdens(C_4) = \frac{3}{2}$.
\end{example}

\section{Propriedade de Ramsey em Grafos Aleatﺣﺏrios}
% ============================================================

\begin{definition}[Propriedade $r$-Ramsey]
Para grafos $G$ e $H$ e um inteiro $r \ge 2$, escrevemos
$G \ramsey{r} H$ para denotar a propriedade de que,
em toda coloraﺣ۶ﺣ۲o das arestas de $G$ com $r$ cores,
existe pelo menos uma cﺣﺏpia monocromﺣ۰tica de $H$ em $G$.
\end{definition}

O seguinte resultado ﺣ۸ o marco fundamental da teoria:

\begin{theorem}[Rﺣﭘdl--Ruciﺧski \cite{rodl1993lower, rodl1994random, rodl1995threshold}]
\label{thm:rodl-rucinski}
Seja $r \ge 2$ um inteiro e seja $H$ um grafo que nﺣ۲o ﺣ۸ uma floresta de estrelas
(e, no caso $r = 2$, tambﺣ۸m nﺣ۲o ﺣ۸ um caminho de comprimento~$3$).
Existem constantes positivas $c = c(H,r)$ e $C = C(H,r)$ tais que
\[
  \lim_{n\to\infty} \Pr\!\bigl[\Gnp \ramsey{r} H\bigr] =
  \begin{cases}
    0 & \text{se } p \le c\, n^{-1/\mdens(H)},\\
    1 & \text{se } p \ge C\, n^{-1/\mdens(H)}.
  \end{cases}
\]
Em particular, $\hat{p}(n) = n^{-1/\mdens(H)}$ ﺣ۸ o limiar semi-agudo para
a propriedade $r$-Ramsey de $H$ em $\Gnp$.
\end{theorem}

Uma prova curta e elegante do Teorema~\ref{thm:rodl-rucinski} foi dada por
Nenadov e Steger \cite{nenadov2016short}, usando a tﺣ۸cnica de
\emph{containers} de hipergrafos.

\section{Propriedade Anti-Ramsey em Grafos Aleatﺣﺏrios} \label{sec:antiramsey}
% ============================================================

\begin{definition}[Propriedade anti-Ramsey]
\label{def:antiramsey}
Dados grafos $G$ e $H$, escrevemos $G \rb H$ para denotar a
\textbf{propriedade anti-Ramsey} de $G$ com respeito a $H$:
para toda coloraﺣ۶ﺣ۲o prﺣﺏpria das arestas de $G$
(com nﺣﭦmero arbitrﺣ۰rio de cores),
existe pelo menos uma cﺣﺏpia \emph{arco-ﺣ­ris} de $H$ em $G$,
isto ﺣ۸, uma cﺣﺏpia de $H$ em que cada aresta recebe uma cor distinta.
\end{definition}

\begin{remark}
A propriedade $G \rb H$ ﺣ۸ monﺣﺏtona crescente em $G$.
Logo, para qualquer grafo fixo $H$, o grafo aleatﺣﺏrio $\Gnp$ admite
um limiar $p_H^{\mathrm{rb}} = p_H^{\mathrm{rb}}(n)$ para essa propriedade
(cf.~Bollobﺣ۰s--Thomason).
\end{remark}

A questﺣ۲o central ﺣ۸ determinar $p_H^{\mathrm{rb}}$ em termos de parﺣ۱metros
combinatﺣﺏrios de $H$.

\subsection{1-Enunciado: limite superior para o limiar}

O primeiro resultado geral foi obtido por Rﺣﭘdl e Tuza~\cite{rodl1992rainbow}
para ciclos, e depois estendido por Kohayakawa, Konstadinidis e Mota:

\begin{theorem}[Kohayakawa--Konstadinidis--Mota \cite{kohayakawa2014anti}]
\label{thm:1statement}
Seja $H$ um grafo qualquer. Existe uma constante $C > 0$ tal que,
se $p = p(n) \ge C\, n^{-1/\mdens(H)}$, entﺣ۲o \whp\
$\Gnp \rb H$.
Em particular, $p_H^{\mathrm{rb}} \le n^{-1/\mdens(H)}$.
\end{theorem}

Em outras palavras, o limiar anti-Ramsey nunca estﺣ۰ acima do limiar Ramsey clﺣ۰ssico.

\subsection{0-Enunciado: limite inferior ﻗ o \emph{framework} de Nenadov et al.}

Para o limite inferior, Nenadov, Person, ﺧ koriﺥ e Steger desenvolveram um
arcabouﺣ۶o geral que reduz o problema probabilﺣ­stico a uma questﺣ۲o determinﺣ­stica:

\begin{theorem}[Nenadov--Person--ﺧ koriﺥ--Steger \cite{nenadov2017algorithmic}]
\label{thm:framework}
Seja $H$ um grafo estritamente $2$-balanceado.
Se todo grafo $G$ tal que $G \rb H$ satisfaz $\maxdens(G) > \mdens(H)$,
entﺣ۲o existe uma constante $c > 0$ tal que, se $p \le c\, n^{-1/\mdens(H)}$,
entﺣ۲o \whp\ $\Gnp \not\rb H$.

Em outras palavras, verificar a condiﺣ۶ﺣ۲o determinﺣ­stica acima
ﺣ۸ suficiente para obter o $0$-enunciado semi-agudo.
\end{theorem}

\begin{definition}[Condiﺣ۶ﺣ۲o determinﺣ­stica anti-Ramsey]
\label{def:cond-det}
Dado um grafo $H$, dizemos que $H$ \textbf{satisfaz a condiﺣ۶ﺣ۲o}~(D)
se todo grafo $G$ tal que $G \rb H$ satisfaz $\maxdens(G) > \mdens(H)$.
\end{definition}

\subsection{Resultados para classes especﺣ­ficas de grafos}

\subsubsection*{Ciclos}

\begin{theorem}[Barros--Cavalar--Mota--Parczyk \cite{AntiCycles}]
\label{thm:anticycles}
O limiar $p_{C_\ell}^{\mathrm{rb}}$ para a propriedade anti-Ramsey do ciclo $C_\ell$
satisfaz:
\begin{itemize}
  \item Se $\ell \ge 5$, entﺣ۲o
    $p_{C_\ell}^{\mathrm{rb}} = n^{-1/\mdens(C_\ell)} = n^{-(\ell-2)/(\ell-1)}$
    (limiar semi-agudo).
  \item Para $\ell = 4$:
    $p_{C_4}^{\mathrm{rb}} = n^{-3/4} \ll n^{-1/\mdens(C_4)} = n^{-2/3}$.
  \item Para $\ell = 3$: a condiﺣ۶ﺣ۲o de ser grafo arco-ﺣ­ris ﺣ۸ satisfeita
    trivialmente, logo
    $p_{C_3}^{\mathrm{rb}} = n^{-1}$ (limiar para a presenﺣ۶a de um triﺣ۱ngulo).
\end{itemize}
\end{theorem}

\subsubsection*{Grafos completos}

\begin{theorem}[Kohayakawa--Mota--Parczyk--Schnitzer \cite{AntiComplete}]
\label{thm:anticomplete}
Para o clique $K_k$, o limiar $p_{K_k}^{\mathrm{rb}}$ satisfaz:
\begin{itemize}
  \item Se $k \ge 5$, entﺣ۲o
    $p_{K_k}^{\mathrm{rb}} = n^{-1/\mdens(K_k)} = n^{-2/(k+1)}$
    (limiar semi-agudo).
  \item Para $k = 4$:
    $p_{K_4}^{\mathrm{rb}} = n^{-7/15} \ll n^{-1/\mdens(K_4)} = n^{-2/5}$.
  \item Para $k = 3$: trivialmente $p_{K_3}^{\mathrm{rb}} = n^{-1}$.
\end{itemize}
\end{theorem}

\subsubsection*{Caso geral: 2-densidade alta}

\begin{theorem}[Kuperwasser \cite{kuperwasser2026anti}]
\label{thm:kuperwasser}
Seja $H$ um grafo com $\mdens(H) \ge 19$.
Existem constantes $c_0 < c_1$ tais que
\[
  \lim_{n\to\infty}\Pr\!\bigl[\Gnp \rb H\bigr] =
  \begin{cases}
    0 & \text{se } p \le c_0\, n^{-1/\mdens(H)},\\
    1 & \text{se } p \ge c_1\, n^{-1/\mdens(H)}.
  \end{cases}
\]
\end{theorem}

O Teorema~\ref{thm:kuperwasser} ﺣ۸ provado verificando a condiﺣ۶ﺣ۲o determinﺣ­stica
do Teorema~\ref{thm:framework} atravﺣ۸s do seguinte resultado:

\begin{theorem}[Kuperwasser \cite{kuperwasser2026anti}]
\label{thm:kuperwasser-det}
Todo grafo $G$ com $\maxdens(G) \ge 18$ admite uma coloraﺣ۶ﺣ۲o prﺣﺏpria das arestas
tal que todo subgrafo arco-ﺣ­ris tem $2$-densidade estritamente menor que $\maxdens(G)$.
Em particular, se $\mdens(H) > 18$, entﺣ۲o $H$ satisfaz a condiﺣ۶ﺣ۲o~\emph{(D)}.
\end{theorem}

\subsection{Questﺣ۲o em aberto: grafos de cintura grande}

Os resultados acima revelam um padrﺣ۲o: o limiar anti-Ramsey diverge de
$n^{-1/\mdens(H)}$ exatamente para grafos ``pequenos'' de cintura baixa
($K_3$, $K_4$, $C_4$). Essa observaﺣ۶ﺣ۲o motiva a seguinte questﺣ۲o central
deste projeto de doutorado:

\begin{question}
\label{q:cintura}
Seja $H$ um grafo com cintura (comprimento do menor ciclo) estritamente maior que~$4$.
ﺣ verdade que $p_H^{\mathrm{rb}} = n^{-1/\mdens(H)}$?
Equivalentemente, $H$ sempre satisfaz a condiﺣ۶ﺣ۲o~\emph{(D)}
(Definiﺣ۶ﺣ۲o~\ref{def:cond-det}) quando $\mathrm{girth}(H) > 4$?
\end{question}

\section{Propriedade Ramsey Restrita (Constrained Ramsey)} \label{sec:cram}
% ============================================================

Este problema, estudado em \cite{behague2025thresholds}, generaliza tanto
a propriedade de Ramsey clﺣ۰ssica quanto a anti-Ramsey.

\begin{definition}[Propriedade Ramsey restrita]
\label{def:cram}
Dados grafos $H_1$ e $H_2$, escrevemos $G \cram (H_1, H_2)$ para denotar
a \textbf{propriedade Ramsey restrita} (\emph{constrained Ramsey}):
para toda coloraﺣ۶ﺣ۲o prﺣﺏpria das arestas de $G$,
existe uma cﺣﺏpia monocromﺣ۰tica de $H_1$ \emph{ou} uma cﺣﺏpia arco-ﺣ­ris de $H_2$.
\end{definition}

\begin{remark}
Quando $H_2$ ﺣ۸ uma floresta, qualquer coloraﺣ۶ﺣ۲o prﺣﺏpria tem trivialmente uma
cﺣﺏpia arco-ﺣ­ris de $H_2$ (desde que o grafo seja grande o suficiente). Logo,
o caso interessante ﺣ۸ quando $H_2$ nﺣ۲o ﺣ۸ uma floresta.

Quando $H_1 = $ estrela com $k$ arestas $K_{1,k}$ e $H_2 = H$, a propriedade
$G \cram (K_{1,k}, H)$ equivale a dizer que toda coloraﺣ۶ﺣ۲o \emph{localmente
$(k-1)$-limitada} de $G$ (em que cada cor aparece no mﺣ۰ximo $k-1$ vezes por vﺣ۸rtice)
contﺣ۸m uma cﺣﺏpia arco-ﺣ­ris de $H$. Assim,
a propriedade anti-Ramsey ﺣ۸ o caso particular $k = 2$
(coloraﺣ۶ﺣﭖes prﺣﺏprias = coloraﺣ۶ﺣﭖes localmente $1$-limitadas).
\end{remark}

\subsection{Resultados principais de Behague et al.}

\begin{theorem}[Behague--Hancock--Hyde--Letzter--Morrison \cite{behague2025thresholds}]
\label{thm:behague-1-stmt}
Sejam $k \ge 2$ e $H$ um grafo qualquer.
Existe uma constante $C > 0$ tal que, se $p \ge C\, n^{-1/\mdens(H)}$, entﺣ۲o
\whp\ $\Gnp \cram (K_{1,k}, H)$.
\end{theorem}

O Teorema~\ref{thm:behague-1-stmt} generaliza o Teorema~\ref{thm:1statement}.

\begin{definition}[Densidades relevantes para Ramsey restrito]
Para o problema restrito com $H_1 = K_{1,k}$, definem-se:
\begin{align*}
  m_a(H) &:= \inf\!\bigl\{\maxdens(G) : G \cram (K_{1,k}, H)\bigr\}\\
         &\phantom{:}= \text{densidade mﺣ۰xima do menor grafo} \ G\ \text{que forﺣ۶a }\\
         &\phantom{= \text{densidade mﺣ۰xima do menor grafo} G \text{que forﺣ۶a}} \text{Ramsey restrito para } (K_{1,k}, H).
\end{align*}
O limiar candidato natural ﺣ۸ entﺣ۲o $n^{-1/m_a(H)}$.
\end{definition}

\begin{theorem}[Behague et al.\ \cite{behague2025thresholds} -- $0$-enunciado geral]
\label{thm:behague-0-stmt}
Sejam $k \ge 3$ e $H$ um grafo que nﺣ۲o ﺣ۸ floresta.
\begin{enumerate}
  \item[(a)] Se $k \ge 4$, \emph{ou} $\mdens(H) \ne 2$, \emph{ou}
    $\mdens(H) = 2$ e $H$ contﺣ۸m um subgrafo estritamente $2$-balanceado
    $J \ne K_3$ com $\mdens(J) = 2$,
    entﺣ۲o existem $c, C > 0$ tais que
    \[
      \lim_{n\to\infty} \Pr\!\bigl[\Gnp \cram (K_{1,k}, H)\bigr] =
      \begin{cases}
        0 & \text{se } p \le c\, n^{-1/\mdens(H)},\\
        1 & \text{se } p \ge C\, n^{-1/\mdens(H)}.
      \end{cases}
    \]
  \item[(b)] Se $k = 3$ e $\mdens(H) = 2$ e o ﺣﭦnico subgrafo estritamente
    $2$-balanceado de $H$ com $2$-densidade~$2$ ﺣ۸ $K_3$ (e.g.\ $H = K_3$),
    entﺣ۲o o limiar ﺣ۸ mais fraco: coarse threshold em $n^{-1/2}$.
\end{enumerate}
\end{theorem}

\section{Propriedade Ramsey Canﺣﺑnica Assimﺣ۸trica} \label{sec:canonico}
% ============================================================

A teoria canﺣﺑnica de Ramsey estuda estruturas que emergem em coloraﺣ۶ﺣﭖes
\emph{arbitrﺣ۰rias} (sem restriﺣ۶ﺣ۲o de nﺣﭦmero de cores).

\begin{definition}[Coloraﺣ۶ﺣ۲o canﺣﺑnica]
\label{def:canonica}
Seja $c : E(G) \to \mathbb{N}$ uma coloraﺣ۶ﺣ۲o arbitrﺣ۰ria das arestas de $G$,
com uma ordenaﺣ۶ﺣ۲o total $v_1 < v_2 < \cdots < v_n$ dos vﺣ۸rtices de $G$.
Um subgrafo $H \subseteq G$ possui uma \textbf{coloraﺣ۶ﺣ۲o canﺣﺑnica} se
satisfaz um dos seguintes padrﺣﭖes:
\begin{enumerate}
  \item \textbf{Monocromﺣ۰tico:} todas as arestas de $H$ tﺣ۹m a mesma cor.
  \item \textbf{Arco-ﺣ­ris:} todas as arestas de $H$ tﺣ۹m cores distintas.
  \item \textbf{Lexicogrﺣ۰fico (min-lex):} a cor de $v_iv_j$ (com $v_i < v_j$)
    depende unicamente de $v_i$.
  \item \textbf{Lexicogrﺣ۰fico (max-lex):} a cor de $v_iv_j$ (com $v_i < v_j$)
    depende unicamente de $v_j$.
\end{enumerate}
\end{definition}

\begin{theorem}[Erdﺧs--Rado \cite{erdos1950combinatorial}]
Para qualquer inteiro $\ell \ge 1$, existe $N = N(\ell)$ tal que toda coloraﺣ۶ﺣ۲o
arbitrﺣ۰ria das arestas de $K_N$ contﺣ۸m uma cﺣﺏpia canﺣﺑnica de $K_\ell$.
\end{theorem}

O resultado de Erdﺧs--Rado foi trazido para o contexto aleatﺣﺏrio por Kamﺥev e Schacht:

\begin{theorem}[Kamﺥev--Schacht \cite{kamcev2023canonical}]
\label{thm:kamcev}
Para todo $\ell \ge 4$, o limiar para a propriedade de que toda coloraﺣ۶ﺣ۲o
arbitrﺣ۰ria de $\Gnp$ contenha uma cﺣﺏpia canﺣﺑnica de $K_\ell$ ﺣ۸
$\hat{p}(n) = n^{-2/(\ell+1)} = n^{-1/\mdens(K_\ell)}$.
\end{theorem}

\subsection{Cenﺣ۰rio assimﺣ۸trico}

\begin{definition}[Propriedade Ramsey canﺣﺑnica assimﺣ۸trica]
\label{def:assimetrico}
Dados grafos $H_1$, $H_2$ e $H_3$, escrevemos
\[
  G \xrightarrow{\mathrm{can}} (H_1, H_2, H_3)
\]
para denotar a seguinte propriedade: para toda coloraﺣ۶ﺣ۲o arbitrﺣ۰ria das
arestas de $G$, existe pelo menos
\begin{itemize}
  \item uma cﺣﺏpia \textbf{monocromﺣ۰tica} de $H_1$, ou
  \item uma cﺣﺏpia \textbf{arco-ﺣ­ris} de $H_2$, ou
  \item uma cﺣﺏpia \textbf{lexicogrﺣ۰fica} de $H_3$.
\end{itemize}
\end{definition}

\begin{question}
\label{q:assimetrico}
Qual ﺣ۸ o comportamento da funﺣ۶ﺣ۲o limiar $\hat{p}(n)$ para a propriedade
$\Gnp \xrightarrow{\mathrm{can}} (H_1, H_2, H_3)$
quando $H_1$, $H_2$ e $H_3$ nﺣ۲o sﺣ۲o todos iguais?
Em particular, como o limiar depende das densidades $\mdens(H_i)$?
\end{question}

% ============================================================
\section{O Caso Assimﺣ۸trico \texorpdfstring{$(K_3, K_4, K_5)$}{(K3, K4, K5)}}
% ============================================================

Queremos determinar o limiar exato para a propriedade $\Gnp \xrightarrow{\mathrm{can}} (K_3,K_4,K_5)$.
Note que, para $p \ll n^{-1/\mdens(K_3)} = n^{-1/2}$, o grafo aleatﺣﺏrio \whp\ nﺣ۲o possui triﺣ۱ngulos, tornando a propriedade trivialmente falsa (atravﺣ۸s de uma coloraﺣ۶ﺣ۲o arco-ﺣ­ris, por exemplo).
Por outro lado, para $p \gg n^{-1/\mdens(K_5)} = n^{-1/3}$, a existﺣ۹ncia de um $K_5$ canﺣﺑnico ﺣ۸ garantida \whp, o que forﺣ۶a a existﺣ۹ncia de uma das estruturas proibidas.

O resultado principal que buscamos formalizar ﺣ۸ que o limiar coincide com o do clique mais denso.
Para a prova do $0$-enunciado, precisaremos das seguintes definiﺣ۶ﺣﭖes estruturais:

\begin{definition}[$H$-fechamento]
\label{def:H-fechado}
Dado um grafo $G$ e um grafo $H$, dizemos que:
\begin{itemize}
  \item Uma aresta $e \in E(G)$ ﺣ۸ \textbf{$H$-fechada} se pertence a pelo menos duas cﺣﺏpias de $H$ em $G$.
  \item Uma cﺣﺏpia $H' \subseteq G$ de $H$ ﺣ۸ \textbf{$H$-fechada} se pelo menos trﺣ۹s de suas arestas sﺣ۲o $H$-fechadas.
  \item O grafo $G$ ﺣ۸ \textbf{$H$-fechado} se todo vﺣ۸rtice e toda aresta de $G$ pertencem a pelo menos uma cﺣﺏpia de $H$, e toda cﺣﺏpia de $H$ em $G$ ﺣ۸ $H$-fechada.
\end{itemize}
\end{definition}

\begin{definition}[$H$-Bloco]
\label{def:H-bloco}
Um grafo $G$ ﺣ۸ um \textbf{$H$-bloco} se $G$ ﺣ۸ $H$-fechado e, para todo subconjunto prﺣﺏprio nﺣ۲o-vazio de arestas $E' \subsetneq E(G)$, existe uma cﺣﺏpia $H'$ de $H$ em $G$ tal que $E(H') \cap E' \ne \emptyset$ e $E(H') \setminus E' \ne \emptyset$.
\end{definition}

\begin{remark}
A chave do framework de Nenadov et al.\ ﺣ۸ que, quando $p \ll n^{-1/\maxdens(H)}$, os $H$-blocos presentes em $\Gnp$ tﺣ۹m tamanho $O(1)$ \whp.
\end{remark}

\begin{theorem}
  Existe um limiar semi-agudo para a propriedade $\Gnp \xrightarrow{\mathrm{can}} (K_3,K_4,K_5)$ em $\hat{p}(n) = n^{-1/3}$. Ou seja, existem constantes $c, C > 0$ tais que:
  \[
  \lim_{n \to \infty} \mathbb{P}(\Gnp \xrightarrow{\mathrm{can}} (K_3,K_4,K_5)) = 
  \begin{cases}
    1 & \text{se } p \ge C n^{-1/3}, \\
    0 & \text{se } p \le c n^{-1/3}.  
  \end{cases}
  \]
\end{theorem}

\begin{proof}[Esboﺣ۶o da Prova]
  O $1$-enunciado segue do Teorema de Kam\v{c}ev--Schacht~\cite{kamcev2023canonical}.

  Para o $0$-enunciado, descrevemos o algoritmo \textsc{Coloraﺣ۶ﺣ۲oCan} que, dado $G$ com $\maxdens(G) \le 3$, constrﺣﺏi explicitamente uma coloraﺣ۶ﺣ۲o das arestas de $G$ sem $K_3$ monocromﺣ۰tico, $K_4$ arco-ﺣ­ris ou $K_5$ lexicogrﺣ۰fico.

  \textbf{Passo 1 --- Coloraﺣ۶ﺣ۲o Lexicogrﺣ۰fica Inicial.}
  Fixamos uma ordenaﺣ۶ﺣ۲o total $\sigma: V(G) \to [n]$ e uma atribuiﺣ۶ﺣ۲o $\varphi: V(G) \to \mathbb{N}$.
  Definimos a coloraﺣ۶ﺣ۲o min-lex $\chi: E(G) \to \mathbb{N}$ por
  \[ \chi(uv) \;:=\; \varphi\!\bigl(\min_\sigma\{u,v\}\bigr), \]
  ou seja, a cor de cada aresta ﺣ۸ determinada pelo vﺣ۸rtice de menor ﺣ­ndice segundo $\sigma$.
  Por construﺣ۶ﺣ۲o, $\chi$ ﺣ۸ uma coloraﺣ۶ﺣ۲o min-lex e satisfaz:
  \begin{itemize}
    \item nﺣ۲o existe $K_3$ monocromﺣ۰tico sob $c$;
    \item nﺣ۲o existe $K_4$ arco-ﺣ­ris sob $c$.
  \end{itemize}
  O problema ﺣ۸ que toda cﺣﺏpia de $K_5$ em $G$ ﺣ۸ lexicogrﺣ۰fica (min-lex) sob $\chi$ --- exatamente a estrutura proibida que precisamos eliminar.

  \textbf{Passo 2 --- Remoﺣ۶ﺣ۲o de Arestas Livres.}
  Removemos de $G$ todas as arestas que nﺣ۲o pertencem a nenhuma cﺣﺏpia de $K_5$.
  Seja $G' \subseteq G$ o grafo resultante. As arestas removidas mantﺣ۹m a coloraﺣ۶ﺣ۲o $\chi$ do Passo~1 e nﺣ۲o causam problemas.

  \textbf{Passo 3 --- Decomposiﺣ۶ﺣ۲o em $K_5$-Blocos.}
  Decompomos $G'$ em seus $K_5$-blocos (Definiﺣ۶ﺣ۲o~\ref{def:H-bloco}).
  Como $\maxdens(G) \le \mdens(K_5) = 3$, cada $K_5$-bloco tem tamanho $O(1)$.

  \textbf{Passo 4 --- Recoloraﺣ۶ﺣ۲o dos Blocos.}
  Para cada $K_5$-bloco $B$, substituﺣ­mos $\chi|_{E(B)}$ por uma nova coloraﺣ۶ﺣ۲o $\chi'$ (construﺣ­da indutivamente) que evita $K_5$ lexicogrﺣ۰fico, sem introduzir $K_3$ monocromﺣ۰tico ou $K_4$ arco-ﺣ­ris.
  A existﺣ۹ncia de tal coloraﺣ۶ﺣ۲o em cada bloco ﺣ۸ garantida pelo Lema~\ref{lem:0-statement-K5} abaixo.
\end{proof}

\begin{lemma}
  \label{lem:0-statement-K5}

  \begin{enumerate}
    \item[(i)] $c$ nﺣ۲o possui $K_3$ monocromﺣ۰tico;
    \item[(ii)] $c$ nﺣ۲o possui $K_4$ arco-ﺣ­ris;
    \item[(iii)] $c$ nﺣ۲o possui $K_5$ lexicogrﺣ۰fico.

\end{lemma}

\medskip
\noindent\textit{Funﺣ۶ﺣ۲o de penalidade e tipos de expansﺣ۲o.}

Para controlar a estrutura dos $K_5$-blocos durante a construﺣ۶ﺣ۲o indutiva,
introduzimos a seguinte medida adaptada de~\cite{AntiComplete}:

\begin{definition}[Funﺣ۶ﺣ۲o de penalidade]
Dado um grafo $G$, definimos
\[
  p(G) \;:=\; e(G) - 3\,v(G) + 5.
\]
\end{definition}

A funﺣ۶ﺣ۲o $p$ ﺣ۸ normalizada de forma que o $K_5$ inicial tenha penalidade nula:
\[
  p(K_5) = 10 - 3 \cdot 5 + 5 = 0.
\]
Alﺣ۸m disso, a condiﺣ۶ﺣ۲o de densidade $\maxdens(G) \le 3$ garante que $e(G) \le 3\,v(G)$, o que implica
$p(G) \le 5$ para qualquer subgrafo de $B$. Essa limitaﺣ۶ﺣ۲o superior ﺣ۸ crucial para controlar a 
complexidade estrutural do bloco.

A construﺣ۶ﺣ۲o de um $K_5$-bloco $B$ pode ser vista como um processo indutivo. Comeﺣ۶amos com uma 
cﺣﺏpia base $H_0 \cong K_5$ e, a cada passo, anexamos novos vﺣ۸rtices e arestas para formar novas 
cﺣﺏpias de $K_5$. Seguindo as estratﺣ۸gias desenvolvidas em~\cite{nenadov2017algorithmic} e \cite{AntiComplete},
classificamos essas extensﺣﭖes em quatro tipos estruturais (ver analogia aos casos de expansﺣ۲o):

\begin{itemize}
  \item \textbf{$R_\ell$ (Expansﺣ۲o regular):} Adicionam-se $3$ novos vﺣ۸rtices e $9$ novas arestas que, juntamente 
    com uma aresta jﺣ۰ existente, formam um novo $K_5$. 
    Neste caso, a variaﺣ۶ﺣ۲o da penalidade ﺣ۸ $\Delta p = 9 - 3(3) = 0$, ou seja, a penalidade do 
    grafo se mantﺣ۸m inalterada.

  \item \textbf{$T_\ell$ (Expansﺣ۲o sobre triﺣ۱ngulo):} Adicionam-se $2$ novos vﺣ۸rtices e $7$ novas arestas que, 
    apoiadas em um triﺣ۱ngulo preexistente, completam um novo $K_5$. 
    Aqui, o incremento na penalidade ﺣ۸ $\Delta p = 7 - 3(2) = 1$.

  \item \textbf{$Q_\ell$ (Expansﺣ۲o sobre $K_4$):} Adiciona-se apenas $1$ novo vﺣ۸rtice ligado a um $K_4$ 
    existente atravﺣ۸s de $4$ novas arestas, completando um $K_5$. 
    O incremento ﺣ۸ $\Delta p = 4 - 3(1) = 1$.

  \item \textbf{$U_\ell$ (Uniﺣ۲o sobre $K_5^-$):} Nﺣ۲o se adicionam vﺣ۸rtices; apenas $1$ nova aresta ﺣ۸ 
    inserida conectando dois vﺣ۸rtices de um $K_5$ quase completo ($K_5 \setminus \{e\}$) jﺣ۰ 
    presente na estrutura. 
    A variaﺣ۶ﺣ۲o ﺣ۸ $\Delta p = 1 - 3(0) = 1$.
\end{itemize}

Uma observaﺣ۶ﺣ۲o fundamental ﺣ۸ que a expansﺣ۲o \textbf{$R_\ell$} (a ﺣﭦnica que ﺣ۸ considerada "regular") 
preserva o valor de $p(G)$. Por outro lado, as expansﺣﭖes degeneradas (\textbf{$T_\ell$}, \textbf{$Q_\ell$} e \textbf{$U_\ell$}) 
aumentam a penalidade estritamente em $1$. Como iniciamos com $p(H_0) = 0$ e sabemos que $p(B) \le 5$, concluﺣ­mos 
que o nﺣﭦmero total de ocorrﺣ۹ncias dos casos $T_\ell$, $Q_\ell$ e $U_\ell$ na construﺣ۶ﺣ۲o do bloco ﺣ۸ uniformemente 
limitado (ocorrem no mﺣ۰ximo $5$ vezes em todo o processo).

\begin{proposition}[Equilﺣ­brio de cores]
\label{prop:equilibrio}
Seja $B$ um $K_5$-bloco com $\maxdens(B) \le 3$ e $\chi'$ a coloraﺣ۶ﺣ۲o auxiliar construﺣ­da
indutivamente sobre as expansﺣﭖes de $B$. Entﺣ۲o ﺣ۸ possﺣ­vel definir $\chi'$ de modo que,
para toda cﺣﺏpia $H \cong K_5$ em $B$ e todo vﺣ۸rtice $w \in V(H)$, $w$ possui ao menos
um vizinho com $\chi'(e) = \text{claro}$ e ao menos um vizinho com $\chi'(e) = \text{escuro}$
dentro de $H$.
\end{proposition}

\noindent
Para provar a Proposiﺣ۶ﺣ۲o~\ref{prop:equilibrio}, estabelecemos duas afirmaﺣ۶ﺣﭖes auxiliares.

\begin{afirmacao}[Expansﺣﭖes de vﺣ۸rtice preservam o equilﺣ­brio]
\label{afirm:vertex-exp}
A propriedade da Proposiﺣ۶ﺣ۲o~\ref{prop:equilibrio} ﺣ۸ preservada em cada passo dos tipos
$R_\ell$, $T_\ell$ e $Q_\ell$.
\end{afirmacao}

\begin{afirmacao}[Limitaﺣ۶ﺣ۲o de $U_\ell$]
\label{afirm:ulimit}
Na construﺣ۶ﺣ۲o de qualquer $K_5$-bloco $B$ com $\maxdens(B) \le 3$,
o nﺣﭦmero de expansﺣﭖes do tipo $U_\ell$ ﺣ۸ no mﺣ۰ximo $2$.
\end{afirmacao}

\begin{proof}[Prova da Proposiﺣ۶ﺣ۲o~\ref{prop:equilibrio}]
Pela Afirmaﺣ۶ﺣ۲o~\ref{afirm:vertex-exp}, a propriedade de equilﺣ­brio ﺣ۸ mantida a cada
expansﺣ۲o dos tipos $R_\ell$, $T_\ell$ e $Q_\ell$.
Pela Afirmaﺣ۶ﺣ۲o~\ref{afirm:ulimit}, o nﺣﭦmero de passos $U_\ell$ ﺣ۸ limitado;
e cada passo $U_\ell$ apenas adiciona uma aresta, sem alterar a propriedade jﺣ۰ garantida para
os trﺣ۹s vizinhos existentes de cada extremidade (como detalhado na prova da Afirmaﺣ۶ﺣ۲o~\ref{afirm:ulimit}).
Portanto, $\chi'$ pode ser construﺣ­da de modo que a propriedade vale apﺣﺏs todos os passos.
\end{proof}

\begin{afirmacao}[$\chi''$ ﺣ۸ uma coloraﺣ۶ﺣ۲o vﺣ۰lida]
\label{afirm:chipp}
Dadas $\chi$ e $\chi'$ como acima, a coloraﺣ۶ﺣ۲o
\[
  \chi'': E(B) \;\longrightarrow\; \mathbb{N} \times \{\text{claro}, \text{escuro}\},
  \qquad \chi''(e) := \bigl(\chi(e),\, \chi'(e)\bigr),
\]
evita $K_3$ monocromﺣ۰tico, $K_4$ arco-ﺣ­ris e $K_5$ lexicogrﺣ۰fico.
Em particular, $\chi''$ ﺣ۸ uma $(K_3, K_4, K_5)$-coloraﺣ۶ﺣ۲o vﺣ۰lida de $B$,
o que conclui a prova do Lema~\ref{lem:0-statement-K5}.
\end{afirmacao}

\begin{proof}
Assumimos que $\chi'$ satisfaz a Proposiﺣ۶ﺣ۲o~\ref{prop:equilibrio} e verificamos cada condiﺣ۶ﺣ۲o
separadamente.

\begin{itemize}
  \item Sob $\chi$, toda cﺣﺏpia de $K_5$ ﺣ۸ lexicogrﺣ۰fica: as quatro arestas $v_1v_j$
  incidentes ao vﺣ۸rtice mﺣ­nimo $v_1$ recebem todas a mesma cor $\chi(v_1v_j) = \varphi(v_1)$.
  Para que o $K_5$ seja tambﺣ۸m min-lex sob $\chi''$, seria necessﺣ۰rio que todas essas
  arestas tivessem o mesmo valor de $\chi'$. Isso ﺣ۸ impossﺣ­vel pela
  Proposiﺣ۶ﺣ۲o~\ref{prop:equilibrio}, que garante que $v_1$ possui ao menos uma aresta
  clara e uma escura dentro de qualquer cﺣﺏpia de $K_5$. Portanto, nﺣ۲o existe
  $K_5$ lexicogrﺣ۰fico sob $\chi''$.

  \item Considere um $K_4$ com vﺣ۸rtice mﺣ­nimo $w$. As trﺣ۹s arestas $wu_i$ incidentes
  a $w$ recebem todas a mesma primeira componente $\chi(wu_i) = \varphi(w)$; para que
  o $K_4$ fosse arco-ﺣ­ris sob $\chi''$, essas trﺣ۹s arestas precisariam ser duas a duas
  distintas na segunda componente $\chi'$. Como sﺣﺏ hﺣ۰ duas cores em $\chi'$, ao menos
  duas delas coincidem em $\chi'$ e, portanto, em $\chi''$. Logo, nﺣ۲o existe
  $K_4$ arco-ﺣ­ris sob $\chi''$.

  \item Em qualquer triﺣ۱ngulo $x < y < z$, as arestas $xy$ e $xz$ recebem $\chi(xy) =
  \varphi(x)$, enquanto $yz$ recebe $\chi(yz) = \varphi(y) \ne \varphi(x)$ (pois $\varphi$
  ﺣ۸ injetiva). Assim, $\chi''(xy) \ne \chi''(yz)$, e o triﺣ۱ngulo nﺣ۲o pode ser
  monocromﺣ۰tico sob $\chi''$.
\end{itemize}
\end{proof}

\noindent
A Afirmaﺣ۶ﺣ۲o~\ref{afirm:chipp} conclui a prova do Lema~\ref{lem:0-statement-K5}.

\end{proof}

\begin{proof}[Prova da Afirmaﺣ۶ﺣ۲o~\ref{afirm:vertex-exp} (Expansﺣﭖes de vﺣ۸rtice)]
A prova ﺣ۸ por induﺣ۶ﺣ۲o no nﺣﭦmero de passos de expansﺣ۲o na construﺣ۶ﺣ۲o do bloco $B$.
Mantemos o seguinte invariante: apﺣﺏs cada passo, toda cﺣﺏpia de $K_5$ presente no grafo
atual admite uma decomposiﺣ۶ﺣ۲o $E(K_5) = E(C_5^{\text{vm}}) \cup E(C_5^{\text{az}})$
(com todo vﺣ۸rtice de grau $2$ em cada $C_5$), e todo triﺣ۱ngulo presente tem exatamente
$2$ arestas de uma cor e $1$ da outra (distribuiﺣ۶ﺣ۲o $2$-$1$).

\medskip

O caso base $H_0 \cong K_5$ satisfaz o invariante pela construﺣ۶ﺣ۲o explﺣ­cita da coloraﺣ۶ﺣ۲o
auxiliar $\chi'$ descrita anteriormente.

\medskip\noindent\textbf{Passo indutivo --- Caso $R_\ell$.}
Uma nova cﺣﺏpia de $K_5$ ﺣ۸ colada ﺣ  estrutura existente por uma ﺣﭦnica aresta $e$ jﺣ۰ colorida
(digamos, $e$ ﺣ۸ vermelha). As $9$ novas arestas ainda nﺣ۲o possuem cor.
Como $K_5$ se decompﺣﭖe em dois $C_5$, e qualquer aresta de $K_5$ pode pertencer ao $C_5$
de qualquer uma das duas cores, basta escolher a decomposiﺣ۶ﺣ۲o do novo $K_5$ de modo que
$e$ pertenﺣ۶a ao $C_5$ vermelho. Com isso, cada vﺣ۸rtice do novo $K_5$ fica com grau $2$
em cada $C_5$, e o invariante ﺣ۸ preservado. Alﺣ۸m disso, como a aresta compartilhada $e$
estﺣ۰ corretamente inserida no $C_5$ da sua cor, nenhum triﺣ۱ngulo preexistente ﺣ۸ afetado.
Note que esta extensﺣ۲o ﺣ۸ sempre possﺣ­vel, independentemente da cor de $e$, e portanto
uma sequﺣ۹ncia arbitrariamente longa de passos $R_\ell$ preserva o invariante.

\medskip\noindent\textbf{Passo indutivo --- Caso $T_\ell$.}
Dois novos vﺣ۸rtices $u, v$ (com a aresta $uv$) sﺣ۲o ligados a um triﺣ۱ngulo $\{a,b,c\}$
preexistente, formando um novo $K_5$ sobre $\{a,b,c,u,v\}$.
Pelo invariante indutivo, o triﺣ۱ngulo $\{a,b,c\}$ possui distribuiﺣ۶ﺣ۲o $2$-$1$;
logo existe um $P_2$ monocromﺣ۰tico, digamos o caminho vermelho $a$--$b$--$c$.
Definimos a coloraﺣ۶ﺣ۲o do novo $K_5$ estendendo esse $P_2$ ao $C_5$ vermelho
$a\text{-}b\text{-}c\text{-}u\text{-}v\text{-}a$; as cinco arestas restantes
$\{au, bu, bv, cv, uv\} \setminus \{av\}$
formam o $C_5$ azul $a\text{-}c\text{-}v\text{-}b\text{-}u\text{-}a$.
Verifica-se diretamente que todo vﺣ۸rtice tem grau $2$ em cada $C_5$, preservando o invariante.

\medskip\noindent\textbf{Passo indutivo --- Caso $Q_\ell$.}
Um novo vﺣ۸rtice $v$ ﺣ۸ ligado a todos os vﺣ۸rtices de um $K_4$ existente $\{a,b,c,d\}$,
formando um novo $K_5$ sobre $\{a,b,c,d,v\}$.
Pelo invariante, toda cﺣﺏpia de $K_5$ que contﺣ۸m o $K_4$ tem distribuiﺣ۶ﺣ۲o $2$-$1$ em seus
triﺣ۱ngulos; em particular, o $K_4$ possui um $P_3$ monocromﺣ۰tico, digamos o caminho
vermelho $a$--$b$--$c$--$d$.
Definimos a coloraﺣ۶ﺣ۲o do novo $K_5$ estendendo esse $P_3$ ao $C_5$ vermelho
$a\text{-}b\text{-}c\text{-}d\text{-}v\text{-}a$; as cinco arestas restantes formam
o $C_5$ azul $a\text{-}c\text{-}v\text{-}b\text{-}d\text{-}a$.
Todo vﺣ۸rtice fica com grau $2$ em cada $C_5$, e o invariante ﺣ۸ preservado.
\end{proof}

\begin{proof}[Prova da Afirmaﺣ۶ﺣ۲o~\ref{afirm:ulimit} (Limitaﺣ۶ﺣ۲o de $U_\ell$)]
Dividimos a prova em dois casos, conforme o bloco $B$ contenha ou nﺣ۲o uma cﺣﺏpia de $K_6$.

\medskip
\noindent\textit{Caso 2: O bloco $B$ contﺣ۸m uma cﺣﺏpia de $K_6$.}

\medskip

Fixamos $H_0 \cong K_6$ como grafo inicial da construﺣ۶ﺣ۲o indutiva de $B$.
Isso ﺣ۸ sempre possﺣ­vel: como $B$ ﺣ۸ um $K_5$-bloco e contﺣ۸m $K_6$, podemos iniciar a
construﺣ۶ﺣ۲o a partir desse $K_6$ e anexar os demais vﺣ۸rtices e arestas de $B$ por
expansﺣﭖes sucessivas, exatamente como nos algoritmos de
\cite{nenadov2017algorithmic} e \cite{AntiComplete}.

\medskip
\noindent\textit{Penalidade inicial.}
Calculamos diretamente:
\[
  p(K_6) \;=\; e(K_6) - 3\,v(K_6) + 5 \;=\; 15 - 3 \cdot 6 + 5 \;=\; 15 - 18 + 5 \;=\; 2.
\]
Como $p(B) \le 5$, o orﺣ۶amento disponﺣ­vel para passos degenerados a partir de $H_0$
ﺣ۸ $5 - 2 = 3$.

\medskip
\noindent\textit{Coloraﺣ۶ﺣ۲o auxiliar de $K_6$.}
Rotulamos os vﺣ۸rtices de $K_6$ como $1, 2, 3, 4, 5, 6$ e definimos $\chi'$ por:
\begin{itemize}
  \item $C_6$ vermelho: $1\text{-}2\text{-}3\text{-}4\text{-}5\text{-}6\text{-}1$
        (as $6$ arestas do ciclo externo);
  \item $C_6$ azul: $1\text{-}3\text{-}5\text{-}1$ e $2\text{-}4\text{-}6\text{-}2$
        reunidos, ou seja, as $6$ arestas $13, 35, 15, 24, 46, 26$
        (o ciclo ``saltando de dois em dois'');
  \item $M_3$ vermelho: o emparelhamento perfeito $\{14, 25, 36\}$.
\end{itemize}
Tem-se $6 + 6 + 3 = 15 = \binom{6}{2}$ arestas, cobrindo exatamente $E(K_6)$.

\medskip
\noindent\textit{Verificaﺣ۶ﺣ۲o das propriedades invariantes.}

A coloraﺣ۶ﺣ۲o acima satisfaz o invariante estabelecido na prova da
Afirmaﺣ۶ﺣ۲o~\ref{afirm:vertex-exp}:

\begin{itemize}
  \item Todo vﺣ۸rtice de $K_6$ tem grau $2$ no $C_6$ vermelho, grau $2$ no $C_6$ azul
        e grau $1$ no $M_3$ vermelho, portanto grau total $3$ em arestas vermelhas e
        grau $2$ em arestas azuis.

  \item Qualquer cﺣﺏpia de $K_5$ em $K_6$ ﺣ۸ obtida removendo um vﺣ۸rtice $w$.
        Dentro desse $K_5$, cada vﺣ۸rtice perde exatamente $1$ vizinho (o prﺣﺏprio $w$),
        mas mantﺣ۸m ao menos $1$ vizinho vermelho e $1$ vizinho azul, preservando o
        equilﺣ­brio da Proposiﺣ۶ﺣ۲o~\ref{prop:equilibrio}.

  \item Todo triﺣ۱ngulo de $K_6$ possui exatamente $2$ arestas vermelhas e $1$ azul,
        ou $1$ vermelha e $2$ azuis (pode-se verificar por inspeﺣ۶ﺣ۲o direta nos
        $\binom{6}{3} = 20$ triﺣ۱ngulos). Em particular, a distribuiﺣ۶ﺣ۲o $2$-$1$ vale,
        garantindo a aplicabilidade da Afirmaﺣ۶ﺣ۲o~\ref{afirm:vertex-exp}.
\end{itemize}

\medskip
\noindent\textit{Como podem surgir passos $U_\ell$ a partir de $K_6$.}

Um passo $U_\ell$ requer a presenﺣ۶a de um $K_5^-$ no grafo atual, isto ﺣ۸, cinco
vﺣ۸rtices faltando exatamente uma aresta para formar $K_5$.
No entanto, dentro de $K_6$ puro, qualquer conjunto de $5$ vﺣ۸rtices induz um $K_5$
completo; logo \emph{nﺣ۲o existe} $K_5^-$ dentro de $H_0 = K_6$.
Portanto, um passo $U_\ell$ nunca pode ser o primeiro passo apﺣﺏs $H_0$: ﺣ۸
necessﺣ۰rio realizar ao menos um passo $T_\ell$ ou $Q_\ell$ antes.

Isso consome ao menos $1$ unidade do orﺣ۶amento de $3$ antes do primeiro $U_\ell$.
Analisamos as duas formas pelas quais um $K_5^-$ pode ser criado:

\begin{itemize}
  \item \textit{Via $Q_\ell$}: adicionamos um novo vﺣ۸rtice $v$ conectado a um $K_4 =
        \{a,b,c,d\} \subset K_6$, deixando $2$ arestas faltando ($ve_1$ e $ve_2$,
        onde $e_1, e_2$ sﺣ۲o os vﺣ۸rtices de $K_6$ fora do $K_4$).
        Surgem imediatamente dois $K_5^-$: $\{a,b,c,e_1,v\}$ faltando $e_1 v$, e
        $\{a,b,c,e_2,v\}$ faltando $e_2 v$.
        Podemos entﺣ۲o realizar dois passos $U_\ell$ consecutivos ($\Delta p = 1$ cada),
        atingindo $p = 2 + 1 + 1 + 1 = 5 \le 5$. Isso totaliza
        \emph{no mﺣ۰ximo $2$ passos $U_\ell$} nesse subcaso.

  \item \textit{Via $T_\ell$}: adicionamos dois novos vﺣ۸rtices $u, v$ ligados a um
        triﺣ۱ngulo $\{a,b,c\} \subset K_6$.
        O vﺣ۸rtice $u$ (e analogamente $v$) nﺣ۲o ﺣ۸ adjacente aos $3$ vﺣ۸rtices restantes
        $d, e, f$ de $K_6$, criando conjuntos $K_5^-$ como $\{a,b,c,d,u\}$ (faltando
        $du$).
        Realizando um passo $U_\ell$ que adiciona $du$, o conjunto $\{a,b,c,d,v\}$
        ainda ﺣ۸ um $K_5^-$ (faltando $dv$), admitindo um segundo $U_\ell$.
        Apﺣﺏs esses dois passos, $p = 2 + 1 + 1 + 1 = 5 \le 5$, e o orﺣ۶amento estﺣ۰
        esgotado.
        Novamente, \emph{no mﺣ۰ximo $2$ passos $U_\ell$} nesse subcaso.
\end{itemize}

Em ambos os subcasos, o total de passos $U_\ell$ quando $H_0 \cong K_6$ ﺣ۸
\emph{no mﺣ۰ximo $2$}, o que conclui o Caso~2.

\end{proof}
