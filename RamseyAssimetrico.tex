\section{Preliminares e NotaĂ§ĂŁo}
% ============================================================

Denotamos por $\Gnp$ o grafo aleatĂłrio binomial com $n$ vĂ©rtices em que cada aresta
estĂˇ presente independentemente com probabilidade $p = p(n)$.
Dizemos que $\Gnp$ satisfaz uma propriedade \emph{assintoticamente quase certamente}
(\whp) se a probabilidade de satisfazĂŞ-la tende a $1$ quando $n \to \infty$.

\begin{definition}[Limiar e tipos de limiar]
Seja $\mathcal{P}$ uma propriedade monĂłtona crescente de grafos.
Uma funĂ§ĂŁo $\hat{p} = \hat{p}(n)$ Ă© um \textbf{limiar grosseiro} (\emph{coarse threshold})
para $\mathcal{P}$ em $\Gnp$ se
\[
  \lim_{n\to\infty} \Pr[\Gnp \in \mathcal{P}] =
  \begin{cases}
    0 & \text{se } p = o(\hat{p}),\\
    1 & \text{se } p = \omega(\hat{p}).
  \end{cases}
\]
Dizemos que $\hat{p}$ Ă© um \textbf{limiar semi-agudo} (\emph{semi-sharp threshold})
se existem constantes $c, C > 0$ tais que
\[
  \lim_{n\to\infty} \Pr[\Gnp \in \mathcal{P}] =
  \begin{cases}
    0 & \text{se } p \le c\,\hat{p},\\
    1 & \text{se } p \ge C\,\hat{p}.
  \end{cases}
\]
O \textbf{$0$-enunciado} refere-se Ă  afirmaĂ§ĂŁo para $p$ abaixo do limiar,
e o \textbf{$1$-enunciado} Ă  afirmaĂ§ĂŁo para $p$ acima do limiar.
\end{definition}

\begin{definition}[Densidades de um grafo]
\label{def:densidades}
Dado um grafo $H$, definem-se:
\begin{itemize}
  \item A \textbf{densidade} de $H$:
    $d(H) := \dfrac{e(H)}{v(H)}$, onde $e(H) = |E(H)|$ e $v(H) = |V(H)|$.

  \item A \textbf{densidade mĂˇxima} de $H$:
    \[
      \maxdens(H) := \max\!\left\{\frac{e(J)}{v(J)} : J \subseteq H,\; v(J) \ge 1\right\}.
    \]
    Dizemos que $H$ Ă© \textbf{balanceado} se $\maxdens(H) = d(H)$, e
    \textbf{estritamente balanceado} se $\maxdens(H) > d(J)$ para todo
    $J \subsetneq H$ com $v(J) \ge 1$.

  \item A \textbf{2-densidade} de $H$ (para $v(H) \ge 3$):
    \[
      d_2(H) := \frac{e(H)-1}{v(H)-2}.
    \]

  \item A \textbf{2-densidade mĂˇxima} de $H$:
    \[
      \mdens(H) := \max\!\left\{d_2(J) : J \subseteq H,\; v(J) \ge 3\right\}
        \cup \left\{\tfrac{1}{2}\right\}.
    \]
    (A fraĂ§ĂŁo $\frac{1}{2}$ corresponde ao caso $H \cong K_2$.)
    Dizemos que $H$ Ă© \textbf{$2$-balanceado} se $\mdens(H) = d_2(H)$, e
    \textbf{estritamente $2$-balanceado} se $\mdens(H) > d_2(J)$ para todo
    $J \subsetneq H$.
\end{itemize}
\end{definition}

\begin{example}
  Para o clique $K_k$ com $k \ge 3$:
  $\mdens(K_k) = \dfrac{\binom{k}{2}-1}{k-2} = \dfrac{k+1}{2}$.
  Para o ciclo $C_\ell$ com $\ell \ge 3$:
  $\mdens(C_\ell) = \dfrac{\ell-1}{\ell-2}$.
  Em particular,
  $\mdens(K_5) = 3$, $\mdens(C_5) = \frac{4}{3}$, $\mdens(C_4) = \frac{3}{2}$.
\end{example}

\section{Propriedade de Ramsey em Grafos AleatĂłrios}
% ============================================================

\begin{definition}[Propriedade $r$-Ramsey]
Para grafos $G$ e $H$ e um inteiro $r \ge 2$, escrevemos
$G \ramsey{r} H$ para denotar a propriedade de que,
em toda coloraĂ§ĂŁo das arestas de $G$ com $r$ cores,
existe pelo menos uma cĂłpia monocromĂˇtica de $H$ em $G$.
\end{definition}

O seguinte resultado Ă© o marco fundamental da teoria:

\begin{theorem}[RĂ¶dl--RuciĹ„ski \cite{rodl1993lower, rodl1994random, rodl1995threshold}]
\label{thm:rodl-rucinski}
Seja $r \ge 2$ um inteiro e seja $H$ um grafo que nĂŁo Ă© uma floresta de estrelas
(e, no caso $r = 2$, tambĂ©m nĂŁo Ă© um caminho de comprimento~$3$).
Existem constantes positivas $c = c(H,r)$ e $C = C(H,r)$ tais que
\[
  \lim_{n\to\infty} \Pr\!\bigl[\Gnp \ramsey{r} H\bigr] =
  \begin{cases}
    0 & \text{se } p \le c\, n^{-1/\mdens(H)},\\
    1 & \text{se } p \ge C\, n^{-1/\mdens(H)}.
  \end{cases}
\]
Em particular, $\hat{p}(n) = n^{-1/\mdens(H)}$ Ă© o limiar semi-agudo para
a propriedade $r$-Ramsey de $H$ em $\Gnp$.
\end{theorem}

Uma prova curta e elegante do Teorema~\ref{thm:rodl-rucinski} foi dada por
Nenadov e Steger \cite{nenadov2016short}, usando a tĂ©cnica de
\emph{containers} de hipergrafos.

\section{Propriedade Anti-Ramsey em Grafos AleatĂłrios} \label{sec:antiramsey}
% ============================================================

\begin{definition}[Propriedade anti-Ramsey]
\label{def:antiramsey}
Dados grafos $G$ e $H$, escrevemos $G \rb H$ para denotar a
\textbf{propriedade anti-Ramsey} de $G$ com respeito a $H$:
para toda coloraĂ§ĂŁo prĂłpria das arestas de $G$
(com nĂşmero arbitrĂˇrio de cores),
existe pelo menos uma cĂłpia \emph{arco-Ă­ris} de $H$ em $G$,
isto Ă©, uma cĂłpia de $H$ em que cada aresta recebe uma cor distinta.
\end{definition}

\begin{remark}
A propriedade $G \rb H$ Ă© monĂłtona crescente em $G$.
Logo, para qualquer grafo fixo $H$, o grafo aleatĂłrio $\Gnp$ admite
um limiar $p_H^{\mathrm{rb}} = p_H^{\mathrm{rb}}(n)$ para essa propriedade
(cf.~BollobĂˇs--Thomason).
\end{remark}

A questĂŁo central Ă© determinar $p_H^{\mathrm{rb}}$ em termos de parĂ˘metros
combinatĂłrios de $H$.

\subsection{1-Enunciado: limite superior para o limiar}

O primeiro resultado geral foi obtido por RĂ¶dl e Tuza~\cite{rodl1992rainbow}
para ciclos, e depois estendido por Kohayakawa, Konstadinidis e Mota:

\begin{theorem}[Kohayakawa--Konstadinidis--Mota \cite{kohayakawa2014anti}]
\label{thm:1statement}
Seja $H$ um grafo qualquer. Existe uma constante $C > 0$ tal que,
se $p = p(n) \ge C\, n^{-1/\mdens(H)}$, entĂŁo \whp\
$\Gnp \rb H$.
Em particular, $p_H^{\mathrm{rb}} \le n^{-1/\mdens(H)}$.
\end{theorem}

Em outras palavras, o limiar anti-Ramsey nunca estĂˇ acima do limiar Ramsey clĂˇssico.

\subsection{0-Enunciado: limite inferior â€” o \emph{framework} de Nenadov et al.}

Para o limite inferior, Nenadov, Person, Ĺ koriÄ‡ e Steger desenvolveram um
arcabouĂ§o geral que reduz o problema probabilĂ­stico a uma questĂŁo determinĂ­stica:

\begin{theorem}[Nenadov--Person--Ĺ koriÄ‡--Steger \cite{nenadov2017algorithmic}]
\label{thm:framework}
Seja $H$ um grafo estritamente $2$-balanceado.
Se todo grafo $G$ tal que $G \rb H$ satisfaz $\maxdens(G) > \mdens(H)$,
entĂŁo existe uma constante $c > 0$ tal que, se $p \le c\, n^{-1/\mdens(H)}$,
entĂŁo \whp\ $\Gnp \not\rb H$.

Em outras palavras, verificar a condiĂ§ĂŁo determinĂ­stica acima
Ă© suficiente para obter o $0$-enunciado semi-agudo.
\end{theorem}

\begin{definition}[CondiĂ§ĂŁo determinĂ­stica anti-Ramsey]
\label{def:cond-det}
Dado um grafo $H$, dizemos que $H$ \textbf{satisfaz a condiĂ§ĂŁo}~(D)
se todo grafo $G$ tal que $G \rb H$ satisfaz $\maxdens(G) > \mdens(H)$.
\end{definition}

\subsection{Resultados para classes especĂ­ficas de grafos}

\subsubsection*{Ciclos}

\begin{theorem}[Barros--Cavalar--Mota--Parczyk \cite{AntiCycles}]
\label{thm:anticycles}
O limiar $p_{C_\ell}^{\mathrm{rb}}$ para a propriedade anti-Ramsey do ciclo $C_\ell$
satisfaz:
\begin{itemize}
  \item Se $\ell \ge 5$, entĂŁo
    $p_{C_\ell}^{\mathrm{rb}} = n^{-1/\mdens(C_\ell)} = n^{-(\ell-2)/(\ell-1)}$
    (limiar semi-agudo).
  \item Para $\ell = 4$:
    $p_{C_4}^{\mathrm{rb}} = n^{-3/4} \ll n^{-1/\mdens(C_4)} = n^{-2/3}$.
  \item Para $\ell = 3$: a condiĂ§ĂŁo de ser grafo arco-Ă­ris Ă© satisfeita
    trivialmente, logo
    $p_{C_3}^{\mathrm{rb}} = n^{-1}$ (limiar para a presenĂ§a de um triĂ˘ngulo).
\end{itemize}
\end{theorem}

\subsubsection*{Grafos completos}

\begin{theorem}[Kohayakawa--Mota--Parczyk--Schnitzer \cite{AntiComplete}]
\label{thm:anticomplete}
Para o clique $K_k$, o limiar $p_{K_k}^{\mathrm{rb}}$ satisfaz:
\begin{itemize}
  \item Se $k \ge 5$, entĂŁo
    $p_{K_k}^{\mathrm{rb}} = n^{-1/\mdens(K_k)} = n^{-2/(k+1)}$
    (limiar semi-agudo).
  \item Para $k = 4$:
    $p_{K_4}^{\mathrm{rb}} = n^{-7/15} \ll n^{-1/\mdens(K_4)} = n^{-2/5}$.
  \item Para $k = 3$: trivialmente $p_{K_3}^{\mathrm{rb}} = n^{-1}$.
\end{itemize}
\end{theorem}

\subsubsection*{Caso geral: 2-densidade alta}

\begin{theorem}[Kuperwasser \cite{kuperwasser2026anti}]
\label{thm:kuperwasser}
Seja $H$ um grafo com $\mdens(H) \ge 19$.
Existem constantes $c_0 < c_1$ tais que
\[
  \lim_{n\to\infty}\Pr\!\bigl[\Gnp \rb H\bigr] =
  \begin{cases}
    0 & \text{se } p \le c_0\, n^{-1/\mdens(H)},\\
    1 & \text{se } p \ge c_1\, n^{-1/\mdens(H)}.
  \end{cases}
\]
\end{theorem}

O Teorema~\ref{thm:kuperwasser} Ă© provado verificando a condiĂ§ĂŁo determinĂ­stica
do Teorema~\ref{thm:framework} atravĂ©s do seguinte resultado:

\begin{theorem}[Kuperwasser \cite{kuperwasser2026anti}]
\label{thm:kuperwasser-det}
Todo grafo $G$ com $\maxdens(G) \ge 18$ admite uma coloraĂ§ĂŁo prĂłpria das arestas
tal que todo subgrafo arco-Ă­ris tem $2$-densidade estritamente menor que $\maxdens(G)$.
Em particular, se $\mdens(H) > 18$, entĂŁo $H$ satisfaz a condiĂ§ĂŁo~\emph{(D)}.
\end{theorem}

\section{Propriedade Ramsey Restrita (Constrained Ramsey)} \label{sec:cram}
% ============================================================

Este problema, estudado em \cite{behague2025thresholds}, generaliza tanto
a propriedade de Ramsey clĂˇssica quanto a anti-Ramsey.

\begin{definition}[Propriedade Ramsey restrita]
\label{def:cram}
Dados grafos $H_1$ e $H_2$, escrevemos $G \cram (H_1, H_2)$ para denotar
a \textbf{propriedade Ramsey restrita} (\emph{constrained Ramsey}):
para toda coloraĂ§ĂŁo prĂłpria das arestas de $G$,
existe uma cĂłpia monocromĂˇtica de $H_1$ \emph{ou} uma cĂłpia arco-Ă­ris de $H_2$.
\end{definition}

\begin{remark}
Quando $H_2$ Ă© uma floresta, qualquer coloraĂ§ĂŁo prĂłpria tem trivialmente uma
cĂłpia arco-Ă­ris de $H_2$ (desde que o grafo seja grande o suficiente). Logo,
o caso interessante Ă© quando $H_2$ nĂŁo Ă© uma floresta.

Quando $H_1 = $ estrela com $k$ arestas $K_{1,k}$ e $H_2 = H$, a propriedade
$G \cram (K_{1,k}, H)$ equivale a dizer que toda coloraĂ§ĂŁo \emph{localmente
$(k-1)$-limitada} de $G$ (em que cada cor aparece no mĂˇximo $k-1$ vezes por vĂ©rtice)
contĂ©m uma cĂłpia arco-Ă­ris de $H$. Assim,
a propriedade anti-Ramsey Ă© o caso particular $k = 2$
(coloraĂ§Ăµes prĂłprias = coloraĂ§Ăµes localmente $1$-limitadas).
\end{remark}

\subsection{Resultados principais de Behague et al.}

\begin{theorem}[Behague--Hancock--Hyde--Letzter--Morrison \cite{behague2025thresholds}]
\label{thm:behague-1-stmt}
Sejam $k \ge 2$ e $H$ um grafo qualquer.
Existe uma constante $C > 0$ tal que, se $p \ge C\, n^{-1/\mdens(H)}$, entĂŁo
\whp\ $\Gnp \cram (K_{1,k}, H)$.
\end{theorem}

O Teorema~\ref{thm:behague-1-stmt} generaliza o Teorema~\ref{thm:1statement}.

\begin{definition}[Densidades relevantes para Ramsey restrito]
Para o problema restrito com $H_1 = K_{1,k}$, definem-se:
\begin{align*}
  m_a(H) &:= \inf\!\bigl\{\maxdens(G) : G \cram (K_{1,k}, H)\bigr\}\\
         &\phantom{:}= \text{densidade mĂˇxima do menor grafo} \ G\ \text{que forĂ§a }\\
         &\phantom{= \text{densidade mĂˇxima do menor grafo} G \text{que forĂ§a}} \text{Ramsey restrito para } (K_{1,k}, H).
\end{align*}
O limiar candidato natural Ă© entĂŁo $n^{-1/m_a(H)}$.
\end{definition}

\begin{theorem}[Behague et al.\ \cite{behague2025thresholds} -- $0$-enunciado geral]
\label{thm:behague-0-stmt}
Sejam $k \ge 3$ e $H$ um grafo que nĂŁo Ă© floresta.
\begin{enumerate}
  \item[(a)] Se $k \ge 4$, \emph{ou} $\mdens(H) \ne 2$, \emph{ou}
    $\mdens(H) = 2$ e $H$ contĂ©m um subgrafo estritamente $2$-balanceado
    $J \ne K_3$ com $\mdens(J) = 2$,
    entĂŁo existem $c, C > 0$ tais que
    \[
      \lim_{n\to\infty} \Pr\!\bigl[\Gnp \cram (K_{1,k}, H)\bigr] =
      \begin{cases}
        0 & \text{se } p \le c\, n^{-1/\mdens(H)},\\
        1 & \text{se } p \ge C\, n^{-1/\mdens(H)}.
      \end{cases}
    \]
  \item[(b)] Se $k = 3$ e $\mdens(H) = 2$ e o Ăşnico subgrafo estritamente
    $2$-balanceado de $H$ com $2$-densidade~$2$ Ă© $K_3$ (e.g.\ $H = K_3$),
    entĂŁo o limiar Ă© mais fraco: coarse threshold em $n^{-1/2}$.
\end{enumerate}
\end{theorem}

\section{Propriedade Ramsey CanĂ´nica AssimĂ©trica} \label{sec:canonico}
% ============================================================

A teoria canĂ´nica de Ramsey estuda estruturas que emergem em coloraĂ§Ăµes
\emph{arbitrĂˇrias} (sem restriĂ§ĂŁo de nĂşmero de cores).

\begin{definition}[ColoraĂ§ĂŁo canĂ´nica]
\label{def:canonica}
Seja $c : E(G) \to \mathbb{N}$ uma coloraĂ§ĂŁo arbitrĂˇria das arestas de $G$,
com uma ordenaĂ§ĂŁo total $v_1 < v_2 < \cdots < v_n$ dos vĂ©rtices de $G$.
Um subgrafo $H \subseteq G$ possui uma \textbf{coloraĂ§ĂŁo canĂ´nica} se
satisfaz um dos seguintes padrĂµes:
\begin{enumerate}
  \item \textbf{MonocromĂˇtico:} todas as arestas de $H$ tĂŞm a mesma cor.
  \item \textbf{Arco-Ă­ris:} todas as arestas de $H$ tĂŞm cores distintas.
  \item \textbf{LexicogrĂˇfico (min-lex):} a cor de $v_iv_j$ (com $v_i < v_j$)
    depende unicamente de $v_i$.
  \item \textbf{LexicogrĂˇfico (max-lex):} a cor de $v_iv_j$ (com $v_i < v_j$)
    depende unicamente de $v_j$.
\end{enumerate}
\end{definition}

\begin{theorem}[ErdĹ‘s--Rado \cite{erdos1950combinatorial}]
Para qualquer inteiro $\ell \ge 1$, existe $N = N(\ell)$ tal que toda coloraĂ§ĂŁo
arbitrĂˇria das arestas de $K_N$ contĂ©m uma cĂłpia canĂ´nica de $K_\ell$.
\end{theorem}

O resultado de ErdĹ‘s--Rado foi trazido para o contexto aleatĂłrio por KamÄŤev e Schacht:

\begin{theorem}[KamÄŤev--Schacht \cite{kamcev2023canonical}]
\label{thm:kamcev}
Para todo $\ell \ge 4$, o limiar para a propriedade de que toda coloraĂ§ĂŁo
arbitrĂˇria de $\Gnp$ contenha uma cĂłpia canĂ´nica de $K_\ell$ Ă©
$\hat{p}(n) = n^{-2/(\ell+1)} = n^{-1/\mdens(K_\ell)}$.
\end{theorem}

\subsection{CenĂˇrio assimĂ©trico}

\begin{definition}[Propriedade Ramsey canĂ´nica assimĂ©trica]
\label{def:assimetrico}
Dados grafos $H_1$, $H_2$ e $H_3$, escrevemos
\[
  G \xrightarrow{\mathrm{can}} (H_1, H_2, H_3)
\]
para denotar a seguinte propriedade: para toda coloraĂ§ĂŁo arbitrĂˇria das
arestas de $G$, existe pelo menos
\begin{itemize}
  \item uma cĂłpia \textbf{monocromĂˇtica} de $H_1$, ou
  \item uma cĂłpia \textbf{arco-Ă­ris} de $H_2$, ou
  \item uma cĂłpia \textbf{lexicogrĂˇfica} de $H_3$.
\end{itemize}
\end{definition}

\begin{question}
\label{q:assimetrico}
Qual Ă© o comportamento da funĂ§ĂŁo limiar $\hat{p}(n)$ para a propriedade
$\Gnp \xrightarrow{\mathrm{can}} (H_1, H_2, H_3)$
quando $H_1$, $H_2$ e $H_3$ nĂŁo sĂŁo todos iguais?
Em particular, como o limiar depende das densidades $\mdens(H_i)$?

\section{O Caso AssimĂ©trico \texorpdfstring{$(K_3, K_4, K_5)$}{(K3, K4, K5)}}
% ============================================================

Queremos determinar o limiar exato para a propriedade $\Gnp \xrightarrow{\mathrm{can}} (K_3,K_4,K_5)$.
Note que, para $p \ll n^{-1/\mdens(K_3)} = n^{-1/2}$, o grafo aleatĂłrio \whp\ nĂŁo possui triĂ˘ngulos, tornando a propriedade trivialmente falsa (atravĂ©s de uma coloraĂ§ĂŁo arco-Ă­ris, por exemplo).
Por outro lado, para $p \gg n^{-1/\mdens(K_5)} = n^{-1/3}$, a existĂŞncia de um $K_5$ canĂ´nico Ă© garantida \whp, o que forĂ§a a existĂŞncia de uma das estruturas proibidas.

O resultado principal que buscamos formalizar Ă© que o limiar coincide com o do clique mais denso.
Para a prova do $0$-enunciado, precisaremos das seguintes definiĂ§Ăµes estruturais:

\begin{definition}[$H$-fechamento]
\label{def:H-fechado}
Dado um grafo $G$ e um grafo $H$, dizemos que:
\begin{itemize}
  \item Uma aresta $e \in E(G)$ Ă© \textbf{$H$-fechada} se pertence a pelo menos duas cĂłpias de $H$ em $G$.
  \item Uma cĂłpia $H' \subseteq G$ de $H$ Ă© \textbf{$H$-fechada} se pelo menos trĂŞs de suas arestas sĂŁo $H$-fechadas.
  \item O grafo $G$ Ă© \textbf{$H$-fechado} se todo vĂ©rtice e toda aresta de $G$ pertencem a pelo menos uma cĂłpia de $H$, e toda cĂłpia de $H$ em $G$ Ă© $H$-fechada.
\end{itemize}
\end{definition}

\begin{definition}[$H$-Bloco]
\label{def:H-bloco}
Um grafo $G$ Ă© um \textbf{$H$-bloco} se $G$ Ă© $H$-fechado e, para todo subconjunto prĂłprio nĂŁo-vazio de arestas $E' \subsetneq E(G)$, existe uma cĂłpia $H'$ de $H$ em $G$ tal que $E(H') \cap E' \ne \emptyset$ e $E(H') \setminus E' \ne \emptyset$.
\end{definition}

\begin{remark}
A chave do framework de Nenadov et al.\ Ă© que, quando $p \ll n^{-1/\maxdens(H)}$, os $H$-blocos presentes em $\Gnp$ tĂŞm tamanho $O(1)$ \whp.
\end{remark}

\begin{theorem}
  Existe um limiar semi-agudo para a propriedade $\Gnp \xrightarrow{\mathrm{can}} (K_3,K_4,K_5)$ em $\hat{p}(n) = n^{-1/3}$. Ou seja, existem constantes $c, C > 0$ tais que:
  \[
  \lim_{n \to \infty} \mathbb{P}(\Gnp \xrightarrow{\mathrm{can}} (K_3,K_4,K_5)) = 
  \begin{cases}
    1 & \text{se } p \ge C n^{-1/3}, \\
    0 & \text{se } p \le c n^{-1/3}.  
  \end{cases}
  \]
\end{theorem}

\begin{proof}[EsboĂ§o da Prova]
  O $1$-enunciado segue do Teorema de Kam\v{c}ev--Schacht~\cite{kamcev2023canonical}.

  Para o $0$-enunciado, descrevemos o algoritmo \textsc{ColoraĂ§ĂŁoCan} que, dado $G$ com $\maxdens(G) \le 3$, constrĂłi explicitamente uma coloraĂ§ĂŁo das arestas de $G$ sem $K_3$ monocromĂˇtico, $K_4$ arco-Ă­ris ou $K_5$ lexicogrĂˇfico.

  \textbf{Passo 1 --- ColoraĂ§ĂŁo LexicogrĂˇfica Inicial.}
  Fixamos uma ordenaĂ§ĂŁo total $\sigma: V(G) \to [n]$ e uma atribuiĂ§ĂŁo $\varphi: V(G) \to \mathbb{N}$.
  Definimos a coloraĂ§ĂŁo min-lex $\chi: E(G) \to \mathbb{N}$ por
  \[ \chi(uv) \;:=\; \varphi\!\bigl(\min_\sigma\{u,v\}\bigr), \]
  ou seja, a cor de cada aresta Ă© determinada pelo vĂ©rtice de menor Ă­ndice segundo $\sigma$.
  Por construĂ§ĂŁo, $\chi$ Ă© uma coloraĂ§ĂŁo min-lex e satisfaz:
  \begin{itemize}
    \item nĂŁo existe $K_3$ monocromĂˇtico sob $c$;
    \item nĂŁo existe $K_4$ arco-Ă­ris sob $c$.
  \end{itemize}
  O problema Ă© que toda cĂłpia de $K_5$ em $G$ Ă© lexicogrĂˇfica (min-lex) sob $\chi$ --- exatamente a estrutura proibida que precisamos eliminar.

  \textbf{Passo 2 --- RemoĂ§ĂŁo de Arestas Livres.}
  Removemos de $G$ todas as arestas que nĂŁo pertencem a nenhuma cĂłpia de $K_5$.
  Seja $G' \subseteq G$ o grafo resultante. As arestas removidas mantĂŞm a coloraĂ§ĂŁo $\chi$ do Passo~1 e nĂŁo causam problemas.

  \textbf{Passo 3 --- DecomposiĂ§ĂŁo em $K_5$-Blocos.}
  Decompomos $G'$ em seus $K_5$-blocos (DefiniĂ§ĂŁo~\ref{def:H-bloco}).
  Como $\maxdens(G) \le \mdens(K_5) = 3$, cada $K_5$-bloco tem tamanho $O(1)$.

  \textbf{Passo 4 --- RecoloraĂ§ĂŁo dos Blocos.}
  Para cada $K_5$-bloco $B$, substituĂ­mos $\chi|_{E(B)}$ por uma nova coloraĂ§ĂŁo $\chi'$ (construĂ­da indutivamente) que evita $K_5$ lexicogrĂˇfico, sem introduzir $K_3$ monocromĂˇtico ou $K_4$ arco-Ă­ris.
  A existĂŞncia de tal coloraĂ§ĂŁo em cada bloco Ă© garantida pelo Lema~\ref{lem:0-statement-K5} abaixo.
\end{proof}

\noindent\textit{FunĂ§ĂŁo de penalidade e tipos de expansĂŁo.}

Para controlar a estrutura dos $K_5$-blocos durante a construĂ§ĂŁo indutiva,
introduzimos a seguinte medida adaptada de~\cite{AntiComplete}:

\begin{definition}[FunĂ§ĂŁo de penalidade]
Dado um grafo $G$, definimos
\[
  p(G) \;:=\; e(G) - 3\,v(G) + 5.
\]
\end{definition}

A funĂ§ĂŁo $p$ Ă© normalizada de forma que o $K_5$ inicial tenha penalidade nula:
\[
  p(K_5) = 10 - 3 \cdot 5 + 5 = 0.
\]
AlĂ©m disso, a condiĂ§ĂŁo de densidade $\maxdens(G) \le 3$ garante que $e(G) \le 3\,v(G)$, o que implica
$p(G) \le 5$ para qualquer subgrafo de $B$. Essa limitaĂ§ĂŁo superior Ă© crucial para controlar a 
complexidade estrutural do bloco.

A construĂ§ĂŁo de um $K_5$-bloco $B$ pode ser vista como um processo indutivo. ComeĂ§amos com uma 
cĂłpia base $H_0 \cong K_5$ e, a cada passo, anexamos novos vĂ©rtices e arestas para formar novas 
cĂłpias de $K_5$. Seguindo as estratĂ©gias desenvolvidas em~\cite{nenadov2017algorithmic} e \cite{AntiComplete},
classificamos essas extensĂµes em quatro tipos estruturais (ver analogia aos casos de expansĂŁo):

\begin{itemize}
  \item \textbf{$R_\ell$ (ExpansĂŁo regular):} Adicionam-se $3$ novos vĂ©rtices e $9$ novas arestas que, juntamente 
    com uma aresta jĂˇ existente, formam um novo $K_5$. 
    Neste caso, a variaĂ§ĂŁo da penalidade Ă© $\Delta p = 9 - 3(3) = 0$, ou seja, a penalidade do 
    grafo se mantĂ©m inalterada.

  \item \textbf{$T_\ell$ (ExpansĂŁo sobre triĂ˘ngulo):} Adicionam-se $2$ novos vĂ©rtices e $7$ novas arestas que, 
    apoiadas em um triĂ˘ngulo preexistente, completam um novo $K_5$. 
    Aqui, o incremento na penalidade Ă© $\Delta p = 7 - 3(2) = 1$.

  \item \textbf{$Q_\ell$ (ExpansĂŁo sobre $K_4$):} Adiciona-se apenas $1$ novo vĂ©rtice ligado a um $K_4$ 
    existente atravĂ©s de $4$ novas arestas, completando um $K_5$. 
    O incremento Ă© $\Delta p = 4 - 3(1) = 1$.

  \item \textbf{$U_\ell$ (UniĂŁo sobre $K_5^-$):} NĂŁo se adicionam vĂ©rtices; apenas $1$ nova aresta Ă© 
    inserida conectando dois vĂ©rtices de um $K_5$ quase completo ($K_5 \setminus \{e\}$) jĂˇ 
    presente na estrutura. 
    A variaĂ§ĂŁo Ă© $\Delta p = 1 - 3(0) = 1$.
\end{itemize}

Uma observaĂ§ĂŁo fundamental Ă© que a expansĂŁo \textbf{$R_\ell$} (a Ăşnica que Ă© considerada "regular")

aumentam a penalidade estritamente em $1$. Como iniciamos com $p(H_0) = 0$ e sabemos que $p(B) \le 5$, concluĂ­mos 
que o nĂşmero total de ocorrĂŞncias dos casos $T_\ell$, $Q_\ell$ e $U_\ell$ na construĂ§ĂŁo do bloco Ă© uniformemente

\begin{proposition}[EquilĂ­brio de cores]
\label{prop:equilibrio}
Seja $B$ um $K_5$-bloco com $\maxdens(B) \le 3$ e $\chi'$ a coloraĂ§ĂŁo auxiliar construĂ­da
indutivamente sobre as expansĂµes de $B$. EntĂŁo Ă© possĂ­vel definir $\chi'$ de modo que,
para toda cĂłpia $H \cong K_5$ em $B$ e todo vĂ©rtice $w \in V(H)$, $w$ possui ao menos
um vizinho com $\chi'(e) = \text{claro}$ e ao menos um vizinho com $\chi'(e) = \text{escuro}$
dentro de $H$.
\end{proposition}

\noindent
Para provar a ProposiĂ§ĂŁo~\ref{prop:equilibrio}, estabelecemos duas afirmaĂ§Ăµes auxiliares.

\begin{afirmacao}[ExpansĂµes de vĂ©rtice preservam o equilĂ­brio]
\label{afirm:vertex-exp}
A propriedade da ProposiĂ§ĂŁo~\ref{prop:equilibrio} Ă© preservada em cada passo dos tipos
$R_\ell$, $T_\ell$ e $Q_\ell$.
\end{afirmacao}

\begin{afirmacao}[LimitaĂ§ĂŁo de $U_\ell$]
\label{afirm:ulimit}
Na construĂ§ĂŁo de qualquer $K_5$-bloco $B$ com $\maxdens(B) \le 3$,
o nĂşmero de expansĂµes do tipo $U_\ell$ Ă© no mĂˇximo $2$.
\end{afirmacao}

\begin{proof}[Prova da ProposiĂ§ĂŁo~\ref{prop:equilibrio}]
Pela AfirmaĂ§ĂŁo~\ref{afirm:vertex-exp}, a propriedade de equilĂ­brio Ă© mantida a cada
expansĂŁo dos tipos $R_\ell$, $T_\ell$ e $Q_\ell$.
Pela AfirmaĂ§ĂŁo~\ref{afirm:ulimit}, o nĂşmero de passos $U_\ell$ Ă© limitado;
e cada passo $U_\ell$ apenas adiciona uma aresta, sem alterar a propriedade jĂˇ garantida para
os trĂŞs vizinhos existentes de cada extremidade (como detalhado na prova da AfirmaĂ§ĂŁo~\ref{afirm:ulimit}).
