\section{Preliminares e NotaûÏûÈo}
% ============================================================

Denotamos por $\Gnp$ o grafo aleatû°rio binomial com $n$ vûˋrtices em que cada aresta
estûÀ presente independentemente com probabilidade $p = p(n)$.
Dizemos que $\Gnp$ satisfaz uma propriedade \emph{assintoticamente quase certamente}
(\whp) se a probabilidade de satisfazûˆ-la tende a $1$ quando $n \to \infty$.

\begin{definition}[Limiar e tipos de limiar]
Seja $\mathcal{P}$ uma propriedade monû°tona crescente de grafos.
Uma funûÏûÈo $\hat{p} = \hat{p}(n)$ ûˋ um \textbf{limiar grosseiro} (\emph{coarse threshold})
para $\mathcal{P}$ em $\Gnp$ se
\[
  \lim_{n\to\infty} \Pr[\Gnp \in \mathcal{P}] =
  \begin{cases}
    0 & \text{se } p = o(\hat{p}),\\
    1 & \text{se } p = \omega(\hat{p}).
  \end{cases}
\]
Dizemos que $\hat{p}$ ûˋ um \textbf{limiar semi-agudo} (\emph{semi-sharp threshold})
se existem constantes $c, C > 0$ tais que
\[
  \lim_{n\to\infty} \Pr[\Gnp \in \mathcal{P}] =
  \begin{cases}
    0 & \text{se } p \le c\,\hat{p},\\
    1 & \text{se } p \ge C\,\hat{p}.
  \end{cases}
\]
O \textbf{$0$-enunciado} refere-se û  afirmaûÏûÈo para $p$ abaixo do limiar,
e o \textbf{$1$-enunciado} û  afirmaûÏûÈo para $p$ acima do limiar.
\end{definition}

\begin{definition}[Densidades de um grafo]
\label{def:densidades}
Dado um grafo $H$, definem-se:
\begin{itemize}
  \item A \textbf{densidade} de $H$:
    $d(H) := \dfrac{e(H)}{v(H)}$, onde $e(H) = |E(H)|$ e $v(H) = |V(H)|$.

  \item A \textbf{densidade mûÀxima} de $H$:
    \[
      \maxdens(H) := \max\!\left\{\frac{e(J)}{v(J)} : J \subseteq H,\; v(J) \ge 1\right\}.
    \]
    Dizemos que $H$ ûˋ \textbf{balanceado} se $\maxdens(H) = d(H)$, e
    \textbf{estritamente balanceado} se $\maxdens(H) > d(J)$ para todo
    $J \subsetneq H$ com $v(J) \ge 1$.

  \item A \textbf{2-densidade} de $H$ (para $v(H) \ge 3$):
    \[
      d_2(H) := \frac{e(H)-1}{v(H)-2}.
    \]

  \item A \textbf{2-densidade mûÀxima} de $H$:
    \[
      \mdens(H) := \max\!\left\{d_2(J) : J \subseteq H,\; v(J) \ge 3\right\}
        \cup \left\{\tfrac{1}{2}\right\}.
    \]
    (A fraûÏûÈo $\frac{1}{2}$ corresponde ao caso $H \cong K_2$.)
    Dizemos que $H$ ûˋ \textbf{$2$-balanceado} se $\mdens(H) = d_2(H)$, e
    \textbf{estritamente $2$-balanceado} se $\mdens(H) > d_2(J)$ para todo
    $J \subsetneq H$.
\end{itemize}
\end{definition}

\begin{example}
  Para o clique $K_k$ com $k \ge 3$:
  $\mdens(K_k) = \dfrac{\binom{k}{2}-1}{k-2} = \dfrac{k+1}{2}$.
  Para o ciclo $C_\ell$ com $\ell \ge 3$:
  $\mdens(C_\ell) = \dfrac{\ell-1}{\ell-2}$.
  Em particular,
  $\mdens(K_5) = 3$, $\mdens(C_5) = \frac{4}{3}$, $\mdens(C_4) = \frac{3}{2}$.
\end{example}

\section{Propriedade de Ramsey em Grafos Aleatû°rios}
% ============================================================

\begin{definition}[Propriedade $r$-Ramsey]
Para grafos $G$ e $H$ e um inteiro $r \ge 2$, escrevemos
$G \ramsey{r} H$ para denotar a propriedade de que,
em toda coloraûÏûÈo das arestas de $G$ com $r$ cores,
existe pelo menos uma cû°pia monocromûÀtica de $H$ em $G$.
\end{definition}

O seguinte resultado ûˋ o marco fundamental da teoria:

\begin{theorem}[RûÑdl--Ruciéski \cite{rodl1993lower, rodl1994random, rodl1995threshold}]
\label{thm:rodl-rucinski}
Seja $r \ge 2$ um inteiro e seja $H$ um grafo que nûÈo ûˋ uma floresta de estrelas
(e, no caso $r = 2$, tambûˋm nûÈo ûˋ um caminho de comprimento~$3$).
Existem constantes positivas $c = c(H,r)$ e $C = C(H,r)$ tais que
\[
  \lim_{n\to\infty} \Pr\!\bigl[\Gnp \ramsey{r} H\bigr] =
  \begin{cases}
    0 & \text{se } p \le c\, n^{-1/\mdens(H)},\\
    1 & \text{se } p \ge C\, n^{-1/\mdens(H)}.
  \end{cases}
\]
Em particular, $\hat{p}(n) = n^{-1/\mdens(H)}$ ûˋ o limiar semi-agudo para
a propriedade $r$-Ramsey de $H$ em $\Gnp$.
\end{theorem}

Uma prova curta e elegante do Teorema~\ref{thm:rodl-rucinski} foi dada por
Nenadov e Steger \cite{nenadov2016short}, usando a tûˋcnica de
\emph{containers} de hipergrafos.

\section{Propriedade Anti-Ramsey em Grafos Aleatû°rios} \label{sec:antiramsey}
% ============================================================

\begin{definition}[Propriedade anti-Ramsey]
\label{def:antiramsey}
Dados grafos $G$ e $H$, escrevemos $G \rb H$ para denotar a
\textbf{propriedade anti-Ramsey} de $G$ com respeito a $H$:
para toda coloraûÏûÈo prû°pria das arestas de $G$
(com nû¤mero arbitrûÀrio de cores),
existe pelo menos uma cû°pia \emph{arco-ûÙris} de $H$ em $G$,
isto ûˋ, uma cû°pia de $H$ em que cada aresta recebe uma cor distinta.
\end{definition}

\begin{remark}
A propriedade $G \rb H$ ûˋ monû°tona crescente em $G$.
Logo, para qualquer grafo fixo $H$, o grafo aleatû°rio $\Gnp$ admite
um limiar $p_H^{\mathrm{rb}} = p_H^{\mathrm{rb}}(n)$ para essa propriedade
(cf.~BollobûÀs--Thomason).
\end{remark}

A questûÈo central ûˋ determinar $p_H^{\mathrm{rb}}$ em termos de parûÂmetros
combinatû°rios de $H$.

\subsection{1-Enunciado: limite superior para o limiar}

O primeiro resultado geral foi obtido por RûÑdl e Tuza~\cite{rodl1992rainbow}
para ciclos, e depois estendido por Kohayakawa, Konstadinidis e Mota:

\begin{theorem}[Kohayakawa--Konstadinidis--Mota \cite{kohayakawa2014anti}]
\label{thm:1statement}
Seja $H$ um grafo qualquer. Existe uma constante $C > 0$ tal que,
se $p = p(n) \ge C\, n^{-1/\mdens(H)}$, entûÈo \whp\
$\Gnp \rb H$.
Em particular, $p_H^{\mathrm{rb}} \le n^{-1/\mdens(H)}$.
\end{theorem}

Em outras palavras, o limiar anti-Ramsey nunca estûÀ acima do limiar Ramsey clûÀssico.

\subsection{0-Enunciado: limite inferior ã o \emph{framework} de Nenadov et al.}

Para o limite inferior, Nenadov, Person, é koriá e Steger desenvolveram um
arcabouûÏo geral que reduz o problema probabilûÙstico a uma questûÈo determinûÙstica:

\begin{theorem}[Nenadov--Person--é koriá--Steger \cite{nenadov2017algorithmic}]
\label{thm:framework}
Seja $H$ um grafo estritamente $2$-balanceado.
Se todo grafo $G$ tal que $G \rb H$ satisfaz $\maxdens(G) > \mdens(H)$,
entûÈo existe uma constante $c > 0$ tal que, se $p \le c\, n^{-1/\mdens(H)}$,
entûÈo \whp\ $\Gnp \not\rb H$.

Em outras palavras, verificar a condiûÏûÈo determinûÙstica acima
ûˋ suficiente para obter o $0$-enunciado semi-agudo.
\end{theorem}

\begin{definition}[CondiûÏûÈo determinûÙstica anti-Ramsey]
\label{def:cond-det}
Dado um grafo $H$, dizemos que $H$ \textbf{satisfaz a condiûÏûÈo}~(D)
se todo grafo $G$ tal que $G \rb H$ satisfaz $\maxdens(G) > \mdens(H)$.
\end{definition}

\subsection{Resultados para classes especûÙficas de grafos}

\subsubsection*{Ciclos}

\begin{theorem}[Barros--Cavalar--Mota--Parczyk \cite{AntiCycles}]
\label{thm:anticycles}
O limiar $p_{C_\ell}^{\mathrm{rb}}$ para a propriedade anti-Ramsey do ciclo $C_\ell$
satisfaz:
\begin{itemize}
  \item Se $\ell \ge 5$, entûÈo
    $p_{C_\ell}^{\mathrm{rb}} = n^{-1/\mdens(C_\ell)} = n^{-(\ell-2)/(\ell-1)}$
    (limiar semi-agudo).
  \item Para $\ell = 4$:
    $p_{C_4}^{\mathrm{rb}} = n^{-3/4} \ll n^{-1/\mdens(C_4)} = n^{-2/3}$.
  \item Para $\ell = 3$: a condiûÏûÈo de ser grafo arco-ûÙris ûˋ satisfeita
    trivialmente, logo
    $p_{C_3}^{\mathrm{rb}} = n^{-1}$ (limiar para a presenûÏa de um triûÂngulo).
\end{itemize}
\end{theorem}

\subsubsection*{Grafos completos}

\begin{theorem}[Kohayakawa--Mota--Parczyk--Schnitzer \cite{AntiComplete}]
\label{thm:anticomplete}
Para o clique $K_k$, o limiar $p_{K_k}^{\mathrm{rb}}$ satisfaz:
\begin{itemize}
  \item Se $k \ge 5$, entûÈo
    $p_{K_k}^{\mathrm{rb}} = n^{-1/\mdens(K_k)} = n^{-2/(k+1)}$

  \item Para $k = 4$:
    $p_{K_4}^{\mathrm{rb}} = n^{-7/15} \ll n^{-1/\mdens(K_4)} = n^{-2/5}$.
  \item Para $k = 3$: trivialmente $p_{K_3}^{\mathrm{rb}} = n^{-1}$.
\end{itemize}
\end{theorem}

\label{thm:kuperwasser}
Seja $H$ um grafo com $\mdens(H) \ge 19$.
Existem constantes $c_0 < c_1$ tais que
\[
  \lim_{n\to\infty}\Pr\!\bigl[\Gnp \rb H\bigr] =
  \begin{cases}
    0 & \text{se } p \le c_0\, n^{-1/\mdens(H)},\\
    1 & \text{se } p \ge c_1\, n^{-1/\mdens(H)}.
  \end{cases}

\end{theorem}

O Teorema~\ref{thm:kuperwasser} ûˋ provado verificando a condiûÏûÈo determinûÙstica
do Teorema~\ref{thm:framework} atravûˋs do seguinte resultado:

\section{Propriedade Ramsey Restrita (Constrained Ramsey)} \label{sec:cram}
% ============================================================

Este problema, estudado em \cite{behague2025thresholds}, generaliza tanto
a propriedade de Ramsey clûÀssica quanto a anti-Ramsey.

\begin{definition}[Propriedade Ramsey restrita]
\label{def:cram}
Dados grafos $H_1$ e $H_2$, escrevemos $G \cram (H_1, H_2)$ para denotar
a \textbf{propriedade Ramsey restrita} (\emph{constrained Ramsey}):
para toda coloraûÏûÈo prû°pria das arestas de $G$,
existe uma cû°pia monocromûÀtica de $H_1$ \emph{ou} uma cû°pia arco-ûÙris de $H_2$.
\end{definition}

\begin{remark}
Quando $H_2$ ûˋ uma floresta, qualquer coloraûÏûÈo prû°pria tem trivialmente uma
cû°pia arco-ûÙris de $H_2$ (desde que o grafo seja grande o suficiente). Logo,
o caso interessante ûˋ quando $H_2$ nûÈo ûˋ uma floresta.

Quando $H_1 = $ estrela com $k$ arestas $K_{1,k}$ e $H_2 = H$, a propriedade
$G \cram (K_{1,k}, H)$ equivale a dizer que toda coloraûÏûÈo \emph{localmente
$(k-1)$-limitada} de $G$ (em que cada cor aparece no mûÀximo $k-1$ vezes por vûˋrtice)
contûˋm uma cû°pia arco-ûÙris de $H$. Assim,
a propriedade anti-Ramsey ûˋ o caso particular $k = 2$
(coloraûÏûçes prû°prias = coloraûÏûçes localmente $1$-limitadas).
\end{remark}

\subsection{Resultados principais de Behague et al.}

\begin{theorem}[Behague--Hancock--Hyde--Letzter--Morrison \cite{behague2025thresholds}]
\label{thm:behague-1-stmt}
Sejam $k \ge 2$ e $H$ um grafo qualquer.
Existe uma constante $C > 0$ tal que, se $p \ge C\, n^{-1/\mdens(H)}$, entûÈo
\whp\ $\Gnp \cram (K_{1,k}, H)$.
\end{theorem}

O Teorema~\ref{thm:behague-1-stmt} generaliza o Teorema~\ref{thm:1statement}.

\begin{definition}[Densidades relevantes para Ramsey restrito]
Para o problema restrito com $H_1 = K_{1,k}$, definem-se:
\begin{align*}
  m_a(H) &:= \inf\!\bigl\{\maxdens(G) : G \cram (K_{1,k}, H)\bigr\}\\
         &\phantom{:}= \text{densidade mûÀxima do menor grafo} \ G\ \text{que forûÏa }\\
         &\phantom{= \text{densidade mûÀxima do menor grafo} G \text{que forûÏa}} \text{Ramsey restrito para } (K_{1,k}, H).
\end{align*}
O limiar candidato natural ûˋ entûÈo $n^{-1/m_a(H)}$.
\end{definition}

\begin{theorem}[Behague et al.\ \cite{behague2025thresholds} -- $0$-enunciado geral]
\label{thm:behague-0-stmt}
Sejam $k \ge 3$ e $H$ um grafo que nûÈo ûˋ floresta.
\begin{enumerate}
  \item[(a)] Se $k \ge 4$, \emph{ou} $\mdens(H) \ne 2$, \emph{ou}
    $\mdens(H) = 2$ e $H$ contûˋm um subgrafo estritamente $2$-balanceado
    $J \ne K_3$ com $\mdens(J) = 2$,
    entûÈo existem $c, C > 0$ tais que
    \[
      \lim_{n\to\infty} \Pr\!\bigl[\Gnp \cram (K_{1,k}, H)\bigr] =
      \begin{cases}
        0 & \text{se } p \le c\, n^{-1/\mdens(H)},\\
        1 & \text{se } p \ge C\, n^{-1/\mdens(H)}.
      \end{cases}
    \]
  \item[(b)] Se $k = 3$ e $\mdens(H) = 2$ e o û¤nico subgrafo estritamente
    $2$-balanceado de $H$ com $2$-densidade~$2$ ûˋ $K_3$ (e.g.\ $H = K_3$),
    entûÈo o limiar ûˋ mais fraco: coarse threshold em $n^{-1/2}$.
\end{enumerate}
\end{theorem}

\section{Propriedade Ramsey CanûÇnica Assimûˋtrica} \label{sec:canonico}
% ============================================================

A teoria canûÇnica de Ramsey estuda estruturas que emergem em coloraûÏûçes
\emph{arbitrûÀrias} (sem restriûÏûÈo de nû¤mero de cores).

\begin{definition}[ColoraûÏûÈo canûÇnica]
\label{def:canonica}
Seja $c : E(G) \to \mathbb{N}$ uma coloraûÏûÈo arbitrûÀria das arestas de $G$,
com uma ordenaûÏûÈo total $v_1 < v_2 < \cdots < v_n$ dos vûˋrtices de $G$.
Um subgrafo $H \subseteq G$ possui uma \textbf{coloraûÏûÈo canûÇnica} se
satisfaz um dos seguintes padrûçes:
\begin{enumerate}
  \item \textbf{MonocromûÀtico:} todas as arestas de $H$ tûˆm a mesma cor.
  \item \textbf{Arco-ûÙris:} todas as arestas de $H$ tûˆm cores distintas.
  \item \textbf{LexicogrûÀfico (min-lex):} a cor de $v_iv_j$ (com $v_i < v_j$)
    depende unicamente de $v_i$.
  \item \textbf{LexicogrûÀfico (max-lex):} a cor de $v_iv_j$ (com $v_i < v_j$)
    depende unicamente de $v_j$.
\end{enumerate}
\end{definition}

\begin{theorem}[Erdés--Rado \cite{erdos1950combinatorial}]
Para qualquer inteiro $\ell \ge 1$, existe $N = N(\ell)$ tal que toda coloraûÏûÈo
arbitrûÀria das arestas de $K_N$ contûˋm uma cû°pia canûÇnica de $K_\ell$.
\end{theorem}

O resultado de Erdés--Rado foi trazido para o contexto aleatû°rio por Kamáev e Schacht:

\begin{theorem}[Kamáev--Schacht \cite{kamcev2023canonical}]
\label{thm:kamcev}
Para todo $\ell \ge 4$, o limiar para a propriedade de que toda coloraûÏûÈo
arbitrûÀria de $\Gnp$ contenha uma cû°pia canûÇnica de $K_\ell$ ûˋ
$\hat{p}(n) = n^{-2/(\ell+1)} = n^{-1/\mdens(K_\ell)}$.
\end{theorem}

\subsection{CenûÀrio assimûˋtrico}

\begin{definition}[Propriedade Ramsey canûÇnica assimûˋtrica]
\label{def:assimetrico}

\begin{itemize}
  \item uma cû°pia \textbf{monocromûÀtica} de $H_1$, ou
  \item uma cû°pia \textbf{arco-ûÙris} de $H_2$, ou
  \item uma cû°pia \textbf{lexicogrûÀfica} de $H_3$.
\end{itemize}
\end{definition}

\end{question}

Queremos determinar o limiar exato para a propriedade $\Gnp \xrightarrow{\mathrm{can}} (K_3,K_4,K_5)$.
Note que, para $p \ll n^{-1/\mdens(K_3)} = n^{-1/2}$, o grafo aleatû°rio \whp\ nûÈo possui triûÂngulos, tornando a propriedade trivialmente falsa (atravûˋs de uma coloraûÏûÈo arco-ûÙris, por exemplo).
Por outro lado, para $p \gg n^{-1/\mdens(K_5)} = n^{-1/3}$, a existûˆncia de um $K_5$ canûÇnico ûˋ garantida \whp, o que forûÏa a existûˆncia de uma das estruturas proibidas.

O resultado principal que buscamos formalizar ûˋ que o limiar coincide com o do clique mais denso.
Para a prova do $0$-enunciado, precisaremos das seguintes definiûÏûçes estruturais:

\begin{definition}[$H$-fechamento]
\label{def:H-fechado}
Dado um grafo $G$ e um grafo $H$, dizemos que:
\begin{itemize}
  \item Uma aresta $e \in E(G)$ ûˋ \textbf{$H$-fechada} se pertence a pelo menos duas cû°pias de $H$ em $G$.
  \item Uma cû°pia $H' \subseteq G$ de $H$ ûˋ \textbf{$H$-fechada} se pelo menos trûˆs de suas arestas sûÈo $H$-fechadas.
  \item O grafo $G$ ûˋ \textbf{$H$-fechado} se todo vûˋrtice e toda aresta de $G$ pertencem a pelo menos uma cû°pia de $H$, e toda cû°pia de $H$ em $G$ ûˋ $H$-fechada.
\end{itemize}
\end{definition}

\begin{definition}[$H$-Bloco]
\label{def:H-bloco}
Um grafo $G$ ûˋ um \textbf{$H$-bloco} se $G$ ûˋ $H$-fechado e, para todo subconjunto prû°prio nûÈo-vazio de arestas $E' \subsetneq E(G)$, existe uma cû°pia $H'$ de $H$ em $G$ tal que $E(H') \cap E' \ne \emptyset$ e $E(H') \setminus E' \ne \emptyset$.
\end{definition}

\begin{remark}
A chave do framework de Nenadov et al.\ ûˋ que, quando $p \ll n^{-1/\maxdens(H)}$, os $H$-blocos presentes em $\Gnp$ tûˆm tamanho $O(1)$ \whp.
\end{remark}

\begin{theorem}
  Existe um limiar semi-agudo para a propriedade $\Gnp \xrightarrow{\mathrm{can}} (K_3,K_4,K_5)$ em $\hat{p}(n) = n^{-1/3}$. Ou seja, existem constantes $c, C > 0$ tais que:
  \[
  \lim_{n \to \infty} \mathbb{P}(\Gnp \xrightarrow{\mathrm{can}} (K_3,K_4,K_5)) = 
  \begin{cases}
    1 & \text{se } p \ge C n^{-1/3}, \\
    0 & \text{se } p \le c n^{-1/3}.  
  \end{cases}
  \]
\end{theorem}

\begin{proof}[Esbo\c{c}o da Prova]
  O $1$-enunciado segue do Teorema de Kam\v{c}ev--Schacht~\cite{kamcev2023canonical}.

  Para o $0$-enunciado, descrevemos o algoritmo \textsc{Colora\c{c}\~{a}oCan} que, dado $G$ com $\maxdens(G) \le 3$, constr\'{o}i explicitamente uma colora\c{c}\~{a}o das arestas de $G$ sem $K_3$ monocrom\'{a}tico, $K_4$ arco-\'{i}ris ou $K_5$ lexicogr\'{a}fico.

  \textbf{Passo 1 --- Colora\c{c}\~{a}o Lexicogr\'{a}fica Inicial.}
  Fixamos uma ordena\c{c}\~{a}o total $\sigma: V(G) \to [n]$ e uma atribui\c{c}\~{a}o $\varphi: V(G) \to \mathbb{N}$.
  Definimos a colora\c{c}\~{a}o min-lex $\chi: E(G) \to \mathbb{N}$ por
  \[ \chi(uv) \;:=\; \varphi\!\bigl(\min_\sigma\{u,v\}\bigr), \]
  ou seja, a cor de cada aresta \'{e} determinada pelo v\'{e}rtice de menor \'{i}ndice segundo $\sigma$.
  Por constru\c{c}\~{a}o, $\chi$ \'{e} uma colora\c{c}\~{a}o min-lex e satisfaz:
  \begin{itemize}
    \item n\~{a}o existe $K_3$ monocrom\'{a}tico sob $c$;
    \item n\~{a}o existe $K_4$ arco-\'{i}ris sob $c$.
  \end{itemize}
  O problema \'{e} que toda c\'{o}pia de $K_5$ em $G$ \'{e} lexicogr\'{a}fica (min-lex) sob $\chi$ --- exatamente a estrutura proibida que precisamos eliminar.

  \textbf{Passo 2 --- Remo\c{c}\~{a}o de Arestas Livres.}
  Removemos de $G$ todas as arestas que n\~{a}o pertencem a nenhuma c\'{o}pia de $K_5$.
  Seja $G' \subseteq G$ o grafo resultante. As arestas removidas mant\^{e}m a colora\c{c}\~{a}o $\chi$ do Passo~1 e n\~{a}o causam problemas.

  \textbf{Passo 3 --- Decomposi\c{c}\~{a}o em $K_5$-Blocos.}
  Decompomos $G'$ em seus $K_5$-blocos (Defini\c{c}\~{a}o~\ref{def:H-bloco}).
  Como $\maxdens(G) \le \mdens(K_5) = 3$, cada $K_5$-bloco tem tamanho $O(1)$.

  \textbf{Passo 4 --- Recolora\c{c}\~{a}o dos Blocos.}
  Para cada $K_5$-bloco $B$, substitu\'{i}mos $\chi|_{E(B)}$ por uma nova colora\c{c}\~{a}o $\chi'$ (constru\'{i}da indutivamente) que evita $K_5$ lexicogr\'{a}fico, sem introduzir $K_3$ monocrom\'{a}tico ou $K_4$ arco-\'{i}ris.
  A exist\^{e}ncia de tal colora\c{c}\~{a}o em cada bloco \'{e} garantida pelo Lema~\ref{lem:0-statement-K5} abaixo.
\end{proof}

\begin{lemma}
  \label{lem:0-statement-K5}
  Seja $B$ um $K_5$-bloco com $\maxdens(B) \le 3$. Ent\~{a}o existe uma colora\c{c}\~{a}o $c: E(B) \to \mathbb{N}$ tal que:
  \begin{enumerate}
    \item[(i)] $c$ n\~{a}o possui $K_3$ monocrom\'{a}tico;
    \item[(ii)] $c$ n\~{a}o possui $K_4$ arco-\'{i}ris;
    \item[(iii)] $c$ n\~{a}o possui $K_5$ lexicogr\'{a}fico.

\end{lemma}

\noindent\textit{Fun\c{c}\~{a}o de penalidade e tipos de expans\~{a}o.}

Para controlar a estrutura dos $K_5$-blocos durante a constru\c{c}\~{a}o indutiva,
introduzimos a seguinte medida adaptada de~\cite{AntiComplete}:

\begin{definition}[Fun\c{c}\~{a}o de penalidade]
Dado um grafo $G$, definimos
\[
  p(G) \;:=\; e(G) - 3\,v(G) + 5.
\]
\end{definition}

A fun\c{c}\~{a}o $p$ \'{e} normalizada de forma que o $K_5$ inicial tenha penalidade nula:
\[
  p(K_5) = 10 - 3 \cdot 5 + 5 = 0.
\]
Al\'{e}m disso, a condi\c{c}\~{a}o de densidade $\maxdens(G) \le 3$ garante que $e(G) \le 3\,v(G)$, o que implica
$p(G) \le 5$ para qualquer subgrafo de $B$. Essa limita\c{c}\~{a}o superior \'{e} crucial para controlar a 
complexidade estrutural do bloco.

A constru\c{c}\~{a}o de um $K_5$-bloco $B$ pode ser vista como um processo indutivo. Come\c{c}amos com uma 
c\'{o}pia base $H_0 \cong K_5$ e, a cada passo, anexamos novos v\'{e}rtices e arestas para formar novas 
c\'{o}pias de $K_5$. Seguindo as estrat\'{e}gias desenvolvidas em~\cite{nenadov2017algorithmic} e \cite{AntiComplete},
classificamos essas extens\~{o}es em quatro tipos estruturais (ver analogia aos casos de expans\~{a}o):

\begin{itemize}
  \item \textbf{$R_\ell$ (Expans\~{a}o regular):} Adicionam-se $3$ novos v\'{e}rtices e $9$ novas arestas que, juntamente 
    com uma aresta j\'{a} existente, formam um novo $K_5$. 
    Neste caso, a varia\c{c}\~{a}o da penalidade \'{e} $\Delta p = 9 - 3(3) = 0$, ou seja, a penalidade do 
    grafo se mant\'{e}m inalterada.

  \item \textbf{$T_\ell$ (Expans\~{a}o sobre tri\^{a}ngulo):} Adicionam-se $2$ novos v\'{e}rtices e $7$ novas arestas que, 
    apoiadas em um tri\^{a}ngulo preexistente, completam um novo $K_5$. 
    Aqui, o incremento na penalidade \'{e} $\Delta p = 7 - 3(2) = 1$.

  \item \textbf{$Q_\ell$ (Expans\~{a}o sobre $K_4$):} Adiciona-se apenas $1$ novo v\'{e}rtice ligado a um $K_4$ 
    existente atrav\'{e}s de $4$ novas arestas, completando um $K_5$. 
    O incremento \'{e} $\Delta p = 4 - 3(1) = 1$.

  \item \textbf{$U_\ell$ (Uni\~{a}o sobre $K_5^-$):} N\~{a}o se adicionam v\'{e}rtices; apenas $1$ nova aresta \'{e} 
    inserida conectando dois v\'{e}rtices de um $K_5$ quase completo ($K_5 \setminus \{e\}$) j\'{a} 
    presente na estrutura. 
    A varia\c{c}\~{a}o \'{e} $\Delta p = 1 - 3(0) = 1$.
\end{itemize}

Uma observa\c{c}\~{a}o fundamental \'{e} que a expans\~{a}o \textbf{$R_\ell$} (a \'{u}nica que \'{e} considerada "regular") 
preserva o valor de $p(G)$. Por outro lado, as expans\~{o}es degeneradas (\textbf{$T_\ell$}, \textbf{$Q_\ell$} e \textbf{$U_\ell$}) 
aumentam a penalidade estritamente em $1$. Como iniciamos com $p(H_0) = 0$ e sabemos que $p(B) \le 5$, conclu\'{i}mos 
que o n\'{u}mero total de ocorr\^{e}ncias dos casos $T_\ell$, $Q_\ell$ e $U_\ell$ na constru\c{c}\~{a}o do bloco \'{e} uniformemente 
limitado (ocorrem no m\'{a}ximo $5$ vezes em todo o processo).

\begin{proposition}[Equil\'{i}brio de cores]
\label{prop:equilibrio}
Seja $B$ um $K_5$-bloco com $\maxdens(B) \le 3$ e $\chi'$ a colora\c{c}\~{a}o auxiliar constru\'{i}da
indutivamente sobre as expans\~{o}es de $B$. Ent\~{a}o \'{e} poss\'{i}vel definir $\chi'$ de modo que,
para toda c\'{o}pia $H \cong K_5$ em $B$ e todo v\'{e}rtice $w \in V(H)$, $w$ possui ao menos
um vizinho com $\chi'(e) = \text{claro}$ e ao menos um vizinho com $\chi'(e) = \text{escuro}$
dentro de $H$.
\end{proposition}

\noindent
Para provar a Proposi\c{c}\~{a}o~\ref{prop:equilibrio}, estabelecemos duas afirma\c{c}\~{o}es auxiliares.

\begin{afirmacao}[Expans\~{o}es de v\'{e}rtice preservam o equil\'{i}brio]
\label{afirm:vertex-exp}
A propriedade da Proposi\c{c}\~{a}o~\ref{prop:equilibrio} \'{e} preservada em cada passo dos tipos
$R_\ell$, $T_\ell$ e $Q_\ell$.
\end{afirmacao}

\begin{afirmacao}[Limita\c{c}\~{a}o de $U_\ell$]
\label{afirm:ulimit}
Na constru\c{c}\~{a}o de qualquer $K_5$-bloco $B$ com $\maxdens(B) \le 3$,
o n\'{u}mero de expans\~{o}es do tipo $U_\ell$ \'{e} no m\'{a}ximo $2$.
\end{afirmacao}

\begin{proof}[Prova da Proposi\c{c}\~{a}o~\ref{prop:equilibrio}]
Pela Afirma\c{c}\~{a}o~\ref{afirm:vertex-exp}, a propriedade de equil\'{i}brio \'{e} mantida a cada
expans\~{a}o dos tipos $R_\ell$, $T_\ell$ e $Q_\ell$.
Pela Afirma\c{c}\~{a}o~\ref{afirm:ulimit}, o n\'{u}mero de passos $U_\ell$ \'{e} limitado;
e cada passo $U_\ell$ apenas adiciona uma aresta, sem alterar a propriedade j\'{a} garantida para
os tr\^{e}s vizinhos existentes de cada extremidade (como detalhado na prova da Afirma\c{c}\~{a}o~\ref{afirm:ulimit}).
Portanto, $\chi'$ pode ser constru\'{i}da de modo que a propriedade vale ap\'{o}s todos os passos.
\end{proof}

\begin{afirmacao}[$\chi''$ \'{e} uma colora\c{c}\~{a}o v\'{a}lida]
\label{afirm:chipp}
Dadas $\chi$ e $\chi'$ como acima, a colora\c{c}\~{a}o
\[
  \chi'': E(B) \;\longrightarrow\; \mathbb{N} \times \{\text{claro}, \text{escuro}\},
  \qquad \chi''(e) := \bigl(\chi(e),\, \chi'(e)\bigr),
\]
evita $K_3$ monocrom\'{a}tico, $K_4$ arco-\'{i}ris e $K_5$ lexicogr\'{a}fico.
Em particular, $\chi''$ \'{e} uma $(K_3, K_4, K_5)$-colora\c{c}\~{a}o v\'{a}lida de $B$,
o que conclui a prova do Lema~\ref{lem:0-statement-K5}.
\end{afirmacao}

\begin{proof}
Supondo que $\chi'$ satisfaz a Proposi\c{c}\~{a}o~\ref{prop:equilibrio}:

\begin{itemize}
  \item \textbf{N\~{a}o existe $K_5$ lexicogr\'{a}fico sob $\chi''$.}
  Sob $\chi$, toda c\'{o}pia de $K_5$ \'{e} lexicogr\'{a}fica: as quatro arestas $v_1v_j$
  incidentes ao v\'{e}rtice m\'{i}nimo $v_1$ t\^{e}m todas $\chi(v_1v_j) = \varphi(v_1)$.
  Para ser min-lex sob $\chi''$, todas essas arestas precisariam ter o mesmo $\chi'$.
  Pela Proposi\c{c}\~{a}o~\ref{prop:equilibrio}, $v_1$ possui ao menos uma aresta clara e uma escura;
  logo o $K_5$ n\~{a}o \'{e} lexicogr\'{a}fico sob $\chi''$.

  \item \textbf{N\~{a}o existe $K_4$ arco-\'{i}ris sob $\chi''$.}
  O v\'{e}rtice m\'{i}nimo $w$ de um $K_4$ tem tr\^{e}s arestas $wu_i$ com $\chi(wu_i) = \varphi(w)$
  (mesma primeira componente). Pela Proposi\c{c}\~{a}o~\ref{prop:equilibrio}, ao menos duas dessas
  arestas coincidem em $\chi'$, logo em $\chi''$; o $K_4$ n\~{a}o \'{e} arco-\'{i}ris.

  \item \textbf{N\~{a}o existe $K_3$ monocrom\'{a}tico sob $\chi''$.}
  Em um tri\^{a}ngulo $x < y < z$, as arestas $xy$ e $yz$ t\^{e}m $\chi(xy) = \varphi(x) \ne \varphi(y) = \chi(yz)$
  (pois $\varphi$ \'{e} injetiva), logo $\chi''(xy) \ne \chi''(yz)$; o tri\^{a}ngulo n\~{a}o \'{e} monocrom\'{a}tico.
\end{itemize}
\end{proof}

\noindent
A Afirma\c{c}\~{a}o~\ref{afirm:chipp} conclui a prova do Lema~\ref{lem:0-statement-K5}.

\end{proof}

\begin{proof}[Prova da Afirma\c{c}\~{a}o~\ref{afirm:vertex-exp} (Expans\~{o}es de v\'{e}rtice)]
Examinamos cada tipo de expans\~{a}o que adiciona v\'{e}rtices, mostrando que a colora\c{c}\~{a}o $\chi'$
pode ser estendida preservando a propriedade de equil\'{i}brio.

\medskip\noindent\textbf{Caso $R_\ell$.}
A nova c\'{o}pia de $K_5$ compartilha uma aresta $e$ j\'{a} colorida.
Constru\'{i}mos os dois $C_5$ do novo $K_5$ de modo que $e$ perten\c{c}a ao $C_5$ da sua cor;
cada v\'{e}rtice fica com grau $2$ em cada $C_5$, garantindo ambas as cores.

\medskip\noindent\textbf{Caso $T_\ell$.}
Um $P_2$ monocrom\'{a}tico $a$--$b$--$c$ (que sempre existe por paridade no tri\^{a}ngulo)
\'{e} extendido ao $C_5$ vermelho $a$--$b$--$c$--$u$--$v$--$a$;
as cinco arestas restantes formam o $C_5$ azul.
Todo v\'{e}rtice fica com grau $2$ em cada $C_5$.

\medskip\noindent\textbf{Caso $Q_\ell$.}
O $P_3$ vermelho $a$--$b$--$c$--$d$ no $K_4$ \'{e} extendido ao $C_5$ vermelho
$a$--$b$--$c$--$d$--$v$--$a$; as cinco arestas restantes formam o $C_5$ azul.
\end{proof}

\begin{proof}[Prova da Afirma\c{c}\~{a}o~\ref{afirm:ulimit} (Limita\c{c}\~{a}o de $U_\ell$)]
Cada passo $U_\ell$ adiciona $1$ aresta e $0$ v\'{e}rtices, aumentando a penalidade em
$\Delta p = 1$. Como $p(H_0) = 0$ e $p(B) \le 5$, podem ocorrer no m\'{a}ximo $5$ passos
degenerados no total ($T_\ell$ + $Q_\ell$ + $U_\ell$). Mostramos que, em qualquer sequ\^{e}ncia de
constru\c{c}\~{a}o de $B$, os passos $U_\ell$ s\~{a}o no m\'{a}ximo $2$.

\emph{[Prova detalhada a completar.]}
\end{proof}
